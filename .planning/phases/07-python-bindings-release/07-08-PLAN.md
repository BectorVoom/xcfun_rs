    ```rust
    //! Topological cargo publish driver (D-15 + Pitfall 4 polling + Pitfall 5 dry-run-first).
    //!
    //! Run modes:
    //!   --dry-run    print the publish order; per-crate `cargo publish --dry-run`; no side effects.
    //!   --execute    after dry-run validates, run `cargo publish` per crate with index propagation polling.
    //!   --from <c>   resume from crate `<c>`; skip earlier crates (idempotent recovery).
    //!   --skip <c>   debug — skip crate `<c>` entirely.
    //!
    //! The driver computes topological order from `cargo metadata`; the expected
    //! result for v0.1.0 is:
    //!   xcfun-ad → xcfun-core → xcfun-kernels → xcfun-eval → xcfun-gpu → xcfun-rs → xcfun-capi
    //!
    //! `xcfun-py` is PyPI-only (NOT published to crates.io). `xtask` and `validation`
    //! are `publish = false`.

    use anyhow::{anyhow, bail, Context, Result};
    use serde_json::Value;
    use std::collections::{HashMap, HashSet};
    use std::process::Command;
    use std::thread;
    use std::time::{Duration, Instant};

    /// Workspace-internal crates that are NEVER published to crates.io.
    const NEVER_PUBLISH: &[&str] = &["xtask", "validation", "xcfun-py", "xcfun-python"];

    /// Polling interval between `cargo search` index-propagation checks.
    const POLL_INTERVAL: Duration = Duration::from_secs(5);
    /// Total cap on index-propagation polling per crate (Pitfall 4).
    const POLL_TIMEOUT: Duration = Duration::from_secs(300);
    /// Cooldown between successive `cargo publish` calls when --execute is set.
    /// 30s is conservative; the cargo search poll loop tightens this dynamically.
    const PUBLISH_COOLDOWN: Duration = Duration::from_secs(30);

    #[derive(Debug, Clone)]
    struct CrateInfo {
        name: String,
        version: String,
        deps_in_workspace: Vec<String>,
    }

    fn main() -> Result<()> {
        let args: Vec<String> = std::env::args().skip(1).collect();
        let mut dry_run = true;
        let mut execute = false;
        let mut from: Option<String> = None;
        let mut skip: Vec<String> = Vec::new();

        let mut iter = args.iter();
        while let Some(arg) = iter.next() {
            match arg.as_str() {
                "--dry-run" => dry_run = true,
                "--execute" => { dry_run = false; execute = true; }
                "--from" => from = iter.next().map(|s| s.to_string()),
                "--skip" => {
                    if let Some(s) = iter.next() { skip.push(s.to_string()); }
                }
                "--help" | "-h" => {
                    print_help();
                    return Ok(());
                }
                _ => bail!("Unknown flag: {arg}. Run --help for usage."),
            }
        }

        if execute && dry_run {
            // The flag combination `--execute --dry-run` IS allowed if the user
            // explicitly typed both — they want execute mode after a dry run.
            // But we treat `--dry-run --execute` as: execute-mode (per the
            // last-flag-wins convention). Here we already set dry_run=false on
            // --execute, so reaching this branch means the user typed --execute then --dry-run.
            bail!("--execute and --dry-run are mutually exclusive");
        }

        let crates = topological_publish_order()?;
        let crates: Vec<CrateInfo> = crates
            .into_iter()
            .filter(|c| !skip.contains(&c.name))
            .collect();

        // Apply --from filter.
        let crates: Vec<CrateInfo> = if let Some(start) = from.as_deref() {
            let pos = crates.iter().position(|c| c.name == start)
                .with_context(|| format!("--from: crate {start} not in publish set"))?;
            crates.into_iter().skip(pos).collect()
        } else {
            crates
        };

        println!("Publish order ({} crates):", crates.len());
        for c in &crates {
            println!("  {} = {}", c.name, c.version);
        }
        println!();

        for c in &crates {
            // Idempotent skip — already on crates.io at this exact version.
            if is_published_at_version(&c.name, &c.version)? {
                println!("[skip] {}@{} already on crates.io", c.name, c.version);
                continue;
            }

            // Pitfall 5 — `cargo publish --dry-run` BEFORE `--execute` cargo publish.
            println!("[dry-run] cargo publish -p {} --dry-run", c.name);
            run(&[
                "cargo", "publish", "-p", &c.name, "--dry-run",
            ])
                .with_context(|| format!("dry-run failed for {}", c.name))?;

            if dry_run {
                println!("[dry-run-only] would publish {}@{}", c.name, c.version);
                continue;
            }

            // --execute path
            println!("[execute] cargo publish -p {}", c.name);
            run(&["cargo", "publish", "-p", &c.name])
                .with_context(|| format!("publish failed for {}", c.name))?;

            // Pitfall 4 — poll cargo search until the new version is visible.
            println!("[wait] polling crates.io index for {}@{} ...", c.name, c.version);
            wait_for_index(&c.name, &c.version)?;

            thread::sleep(PUBLISH_COOLDOWN);
        }

        println!();
        println!("All Rust crates published. Now run (locally or in CI release.yml):");
        println!("    cd crates/xcfun-py && maturin publish --skip-existing");
        Ok(())
    }

    fn print_help() {
        println!("xtask release-publish — topological cargo publish driver (D-15)");
        println!();
        println!("USAGE:");
        println!("    cargo run -p xtask --bin release-publish [-- FLAGS]");
        println!();
        println!("FLAGS:");
        println!("    --dry-run         (default) print order + per-crate cargo publish --dry-run; no side effects");
        println!("    --execute         actually publish; runs --dry-run first per crate, then `cargo publish`");
        println!("    --from <crate>    resume from <crate>; idempotent recovery for partially-published runs");
        println!("    --skip <crate>    skip <crate> entirely (debug only)");
        println!("    --help            this message");
    }

    /// Compute the topological publish order from `cargo metadata`.
    fn topological_publish_order() -> Result<Vec<CrateInfo>> {
        let out = Command::new("cargo")
            .args(["metadata", "--format-version", "1", "--no-deps"])
            .output()
            .context("invoke cargo metadata")?;
        if !out.status.success() {
            bail!("cargo metadata failed: {}", String::from_utf8_lossy(&out.stderr));
        }
        let v: Value = serde_json::from_slice(&out.stdout)
            .context("parse cargo metadata JSON")?;

        let workspace_members: HashSet<String> = v["workspace_members"]
            .as_array()
            .ok_or_else(|| anyhow!("missing workspace_members"))?
            .iter()
            .filter_map(|s| s.as_str().map(|x| x.split_whitespace().next().unwrap_or("").to_string()))
            .collect();

        // Re-run with --no-deps OFF to get inter-package dep edges.
        let out2 = Command::new("cargo")
            .args(["metadata", "--format-version", "1"])
            .output()
            .context("invoke cargo metadata (with deps)")?;
        let v2: Value = serde_json::from_slice(&out2.stdout)
            .context("parse cargo metadata JSON (with deps)")?;

        let packages = v2["packages"].as_array()
            .ok_or_else(|| anyhow!("missing packages"))?;

        let mut crates: HashMap<String, CrateInfo> = HashMap::new();
        for pkg in packages {
            let name = pkg["name"].as_str()
                .ok_or_else(|| anyhow!("package missing name"))?
                .to_string();
            if !workspace_members.contains(&name) { continue; }
            if NEVER_PUBLISH.contains(&name.as_str()) { continue; }
            // Filter publish=false entries (xtask sets `publish = false`).
            if pkg.get("publish") == Some(&Value::Array(vec![])) {
                // `publish = false` in Cargo.toml ⇒ "publish": [] in metadata.
                continue;
            }
            let version = pkg["version"].as_str()
                .ok_or_else(|| anyhow!("{name}: missing version"))?
                .to_string();
            let deps: Vec<String> = pkg["dependencies"].as_array()
                .map(|arr| arr.iter()
                    .filter_map(|d| d["name"].as_str().map(|s| s.to_string()))
                    .filter(|d| workspace_members.contains(d)
                        && !NEVER_PUBLISH.contains(&d.as_str()))
                    .collect())
                .unwrap_or_default();
            crates.insert(name.clone(), CrateInfo {
                name,
                version,
                deps_in_workspace: deps,
            });
        }

        // Kahn's algorithm.
        let mut in_degree: HashMap<String, usize> = crates.iter()
            .map(|(k, v)| (k.clone(), v.deps_in_workspace.len()))
            .collect();
        let mut order = Vec::new();
        loop {
            let next = crates.values()
                .filter(|c| in_degree.get(&c.name).copied() == Some(0))
                .min_by_key(|c| c.name.clone())  // deterministic tie-break
                .map(|c| c.name.clone());
            let Some(name) = next else { break };
            in_degree.remove(&name);
            order.push(crates[&name].clone());
            for c in crates.values() {
                if c.deps_in_workspace.contains(&name) {
                    if let Some(d) = in_degree.get_mut(&c.name) {
                        *d = d.saturating_sub(1);
                    }
                }
            }
        }
        if !in_degree.is_empty() {
            bail!("cyclic publish graph: {:?}", in_degree.keys().collect::<Vec<_>>());
        }
        Ok(order)
    }

    fn is_published_at_version(name: &str, version: &str) -> Result<bool> {
        let out = Command::new("cargo")
            .args(["search", name, "--limit", "1"])
            .output()
            .context("cargo search")?;
        if !out.status.success() {
            // Network failure — treat as "not yet published" + warn; the user
            // can retry or pass --skip.
            eprintln!("[warn] cargo search failed for {name}; assuming not-yet-published");
            return Ok(false);
        }
        let stdout = String::from_utf8_lossy(&out.stdout);
        // cargo search prints `name = "version"  # description`
        let needle = format!("{name} = \"{version}\"");
        Ok(stdout.contains(&needle))
    }

    fn wait_for_index(name: &str, version: &str) -> Result<()> {
        let start = Instant::now();
        while start.elapsed() < POLL_TIMEOUT {
            if is_published_at_version(name, version)? {
                println!("[ok] {name}@{version} visible on index");
                return Ok(());
            }
            thread::sleep(POLL_INTERVAL);
        }
        bail!("timed out waiting for {name}@{version} index propagation (>{POLL_TIMEOUT:?})")
    }

    fn run(args: &[&str]) -> Result<()> {
        let status = Command::new(args[0])
            .args(&args[1..])
            .status()
            .with_context(|| format!("spawn {:?}", args))?;
        if !status.success() {
            bail!("command {:?} exited with {}", args, status);
        }
        Ok(())
    }
    ```
