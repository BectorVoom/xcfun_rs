        "XC_SLATERX", "XC_VWN3C", "XC_VWN5C", "XC_PW92C", "XC_PZ81C",

    /// Emit `validation/c_stubs.cpp` with stub FUNCTIONAL macros for every non-LDA
    /// functional ID. Required for `xcint.cpp`'s template recursion to link in the
    /// validation cc-build (Plan 02-06 Wave-2-2).
    fn emit_c_stubs_cpp(all_functionals: &[Functional]) -> String {
        let mut out = String::new();
        out.push_str("// validation/c_stubs.cpp — AUTO-GENERATED by xtask regen-registry. DO NOT EDIT.\n");
        out.push_str("// Stubs for Phase 2 cc-compile — every non-LDA functional ID needs a fundat_db\n");
        out.push_str("// specialisation or xcint.cpp template recursion fails to link.\n");
        out.push_str("//\n");
        out.push_str("// Phase 3+ extends this file by re-running regen-registry; LDA IDs already get\n");
        out.push_str("// the real ENERGY_FUNCTION via their respective .cpp files compiled by build.rs.\n\n");
        out.push_str("#include \"functional.hpp\"\n\n");
        out.push_str("template <typename num> static num stub_unimpl(const densvars<num> &) { return num(0); }\n\n");

        for f in all_functionals {
            if PHASE2_LDA_IDS.contains(&f.id.as_str()) {
                continue;  // LDA IDs get their real ENERGY_FUNCTION from the .cpp files
            }
            // Reconstruct the depends bitmask in C++ token form (XC_DENSITY|XC_GRADIENT|...)
            let depends_str = format_depends_bitmask(f.depends);
            out.push_str(&format!(
                "FUNCTIONAL({}) = {{\"stub\", \"stub\", {}, ENERGY_FUNCTION(stub_unimpl)}};\n",
                f.id, depends_str
            ));
        }
        out
    }

    fn format_depends_bitmask(bits: u32) -> String {
        let mut parts = Vec::new();
        if bits & 1 != 0 { parts.push("XC_DENSITY"); }
        if bits & 2 != 0 { parts.push("XC_GRADIENT"); }
        if bits & 4 != 0 { parts.push("XC_LAPLACIAN"); }
        if bits & 8 != 0 { parts.push("XC_KINETIC"); }
        if bits & 16 != 0 { parts.push("XC_JP"); }
        if parts.is_empty() {
            "XC_DENSITY".into()  // Default — never zero
        } else {
            parts.join("|")
        }
    }

    //!
    //! Source: `xcfun-master/api/xcfun.h:1-130` (every `XCFun_API` declaration).

    unsafe extern "C" {
        pub fn xcfun_new() -> *mut c_void;
        pub fn xcfun_delete(fun: *mut c_void);
        pub fn xcfun_set(fun: *mut c_void, name: *const i8, value: f64) -> i32;
        pub fn xcfun_eval_setup(fun: *mut c_void, vars: u32, mode: u32, order: i32) -> i32;
        pub fn xcfun_input_length(fun: *const c_void) -> i32;
        pub fn xcfun_output_length(fun: *const c_void) -> i32;
        pub fn xcfun_eval(fun: *const c_void, density: *const f64, result: *mut f64);
    }

    /// RAII wrapper around the C `xcfun*` opaque handle. `Drop` calls `xcfun_delete`.
    pub struct CppXcfun {
        handle: *mut c_void,
    }

    /// `Send` / `Sync` are NOT auto-implemented (raw pointer). The xcfun-master C++
    /// reference is single-threaded per-handle; do NOT share `CppXcfun` across threads.
    /// Each driver loop iteration creates its own `CppXcfun`.

    impl CppXcfun {
        /// Create a new C-side xcfun handle.
        pub fn new() -> Self {
            let handle = unsafe { xcfun_new() };
            assert!(!handle.is_null(), "xcfun_new returned null");
            Self { handle }
        }

        /// `xcfun_set(name, value)` — activates the named functional with the given weight.
        /// Returns the i32 status code from C++ (0 = success, non-zero = error).
        pub fn set(&mut self, name: &str, value: f64) -> i32 {
            let cname = CString::new(name).expect("name contained NUL byte");
            unsafe { xcfun_set(self.handle, cname.as_ptr(), value) }
        }

        /// `xcfun_eval_setup(vars, mode, order)`.
        pub fn eval_setup(&mut self, vars: u32, mode: u32, order: i32) -> i32 {
            unsafe { xcfun_eval_setup(self.handle, vars, mode, order) }
        }

        /// `xcfun_input_length(fun)`.
        pub fn input_length(&self) -> usize {
            unsafe { xcfun_input_length(self.handle) as usize }
        }

        /// `xcfun_output_length(fun)`.
        pub fn output_length(&self) -> usize {
            unsafe { xcfun_output_length(self.handle) as usize }
        }

        /// `xcfun_eval(input, output)` — assumes input.len() == input_length()
        /// and output.len() == output_length(); panics otherwise.
        pub fn eval(&self, input: &[f64], output: &mut [f64]) {
            let il = self.input_length();
            let ol = self.output_length();
            assert_eq!(input.len(), il, "FFI eval: input.len() mismatch");
            assert_eq!(output.len(), ol, "FFI eval: output.len() mismatch");
            unsafe { xcfun_eval(self.handle, input.as_ptr(), output.as_mut_ptr()) };
        }
    }

        fn drop(&mut self) {
            if !self.handle.is_null() {
                unsafe { xcfun_delete(self.handle) };
            }
        }
    }

    4. Apply per-functional D-24 threshold: 1e-7 for XC_LDAERFX/XC_LDAERFC/XC_LDAERFC_JT; 1e-12 otherwise

    Sketch:
    ```rust
    use anyhow::{Context, Result};
    use std::collections::HashMap;
    use xcfun_core::{taylorlen, FunctionalId, Mode, Vars, FUNCTIONAL_DESCRIPTORS, VARS_TABLE};
    use xcfun_eval::Functional;

    use crate::ffi::CppXcfun;
    use crate::fixtures::GridPoint;

    /// One per (functional, vars, mode, order, point_idx, element_idx) tuple.
    #[derive(serde::Serialize, Debug, Clone)]
    pub struct ReportRecord {
        pub functional: String,
        pub vars: String,
        pub mode: u32,
        pub order: u32,
        pub point_idx: usize,
        pub element_idx: usize,
        pub input: Vec<f64>,
        pub rust: f64,
        pub cpp: f64,
        pub abs_err: f64,
        pub rel_err: f64,
        pub threshold: f64,
        pub pass: bool,
    }

    pub struct Report {
        pub records: Vec<ReportRecord>,
        /// Max rel-error per (functional, order) cell.
        pub matrix: HashMap<(String, u32), CellSummary>,
    }

    #[derive(Debug, Clone)]
    pub struct CellSummary {
        pub max_rel_err: f64,
        pub threshold: f64,
        pub records_total: usize,
        pub records_failed: usize,
    }

    impl Report {
        pub fn new() -> Self {
            Self { records: Vec::new(), matrix: HashMap::new() }
        }

        pub fn push(&mut self, rec: ReportRecord) {
            let key = (rec.functional.clone(), rec.order);
            let entry = self.matrix.entry(key).or_insert(CellSummary {
                max_rel_err: 0.0,
                threshold: rec.threshold,
                records_total: 0,
                records_failed: 0,
            });
            entry.max_rel_err = entry.max_rel_err.max(rec.rel_err);
            entry.records_total += 1;
            if !rec.pass {
                entry.records_failed += 1;
            }
            // Only keep failed records in the JSONL to bound size; passing records
            // contribute to the matrix but don't bloat the JSONL.
            if !rec.pass {
                self.records.push(rec);
            } else if rec.point_idx == 0 && rec.element_idx == 0 {
                // Sample a few passing records per (functional, order) for transparency.
                self.records.push(rec);
            }
        }

        pub fn failed_count(&self) -> usize {
            self.matrix.values().map(|c| c.records_failed).sum()
        }

        pub fn total_records(&self) -> usize {
            self.matrix.values().map(|c| c.records_total).sum()
        }
    }

    /// Per-functional tier-2 threshold (D-24).
    fn threshold_for(name: &str) -> f64 {
        if name.starts_with("XC_LDAERF") {
            1e-7   // D-24 (USER-APPROVED 2026-04-20)
        } else {
            1e-12  // strict per ROADMAP Phase 2 SC #5
        }
    }

    /// Build the per-point input array for a given Vars.
    /// Returns the f64 input slice the kernel expects.
    fn build_input(gp: &GridPoint, vars: Vars) -> Vec<f64> {
        let inlen = VARS_TABLE[vars as usize].len as usize;
        let mut input = vec![0.0_f64; inlen];
        match vars {
            Vars::A_B => {
                let (a, b) = gp.ab_from_ns();
                input[0] = a;
                input[1] = b;
            }
            Vars::A_B_GAA_GAB_GBB => {
                let (a, b) = gp.ab_from_ns();
                input[0] = a;
                input[1] = b;
                input[2] = gp.gaa;
                input[3] = gp.gab;
                input[4] = gp.gbb;
            }
            // Phase 2 LDAs use only the above two vars; other arms unsupported in Phase 2.
            _ => panic!("validation driver: unsupported vars {:?} in Phase 2", vars),
        }
        input
    }

    /// Run tier-2 parity for all 11 Phase-2 LDA functionals against C++ at the given max order.
    pub fn run(grid: &[GridPoint], max_order: u32, filter: &regex::Regex) -> Result<Report> {
        let mut report = Report::new();

        // Iterate over FUNCTIONAL_DESCRIPTORS — but Phase 2 only handles LDA.
        // For LDA-09 part 2 (TW) and LDA-10 (VWK), test_vars is None in the registry
        // (no upstream test_in/out), so we hard-code their vars here.
        let lda_targets: &[(FunctionalId, &str, Vars)] = &[
            (FunctionalId::XC_SLATERX, "XC_SLATERX", Vars::A_B),
            (FunctionalId::XC_VWN3C, "XC_VWN3C", Vars::A_B),
            (FunctionalId::XC_VWN5C, "XC_VWN5C", Vars::A_B),
            (FunctionalId::XC_PW92C, "XC_PW92C", Vars::A_B),
            (FunctionalId::XC_PZ81C, "XC_PZ81C", Vars::A_B),
            (FunctionalId::XC_LDAERFX, "XC_LDAERFX", Vars::A_B),
            (FunctionalId::XC_LDAERFC, "XC_LDAERFC", Vars::A_B),
            (FunctionalId::XC_LDAERFC_JT, "XC_LDAERFC_JT", Vars::A_B),
            (FunctionalId::XC_TFK, "XC_TFK", Vars::A_B),
            (FunctionalId::XC_TW, "XC_TW", Vars::A_B_GAA_GAB_GBB),
            (FunctionalId::XC_VWK, "XC_VWK", Vars::A_B_GAA_GAB_GBB),
        ];

        for &(id, name, vars) in lda_targets {
            if !filter.is_match(&name.to_lowercase()) { continue; }
            tracing::info!("Tier-2: {}", name);
            let threshold = threshold_for(name);
            let inlen = VARS_TABLE[vars as usize].len as usize;

            for order in 0..=max_order.min(2) {
                let outlen = taylorlen(inlen, order as usize);

                // Set up the C++ side once per (functional, order).
                let mut cpp = CppXcfun::new();
                cpp.set(&name.to_lowercase().replace("xc_", ""), 1.0);  // xcfun_set takes lowercase, no XC_ prefix
                let setup = cpp.eval_setup(vars as u32, 1 /* XC_PARTIAL_DERIVATIVES */, order as i32);
                if setup != 0 {
                    anyhow::bail!("xcfun_eval_setup({}, {:?}, order={}) failed: status={}", name, vars, order, setup);
                }
                let cpp_inlen = cpp.input_length();
                let cpp_outlen = cpp.output_length();
                if cpp_inlen != inlen || cpp_outlen != outlen {
                    anyhow::bail!("Length mismatch: rust inlen={} outlen={}; cpp inlen={} outlen={}", inlen, outlen, cpp_inlen, cpp_outlen);
                }

                // Set up Rust side.
                let weights: &'static [(FunctionalId, f64)] = Box::leak(Box::new([(id, 1.0)]));
                // ^ leak is acceptable in the validation crate (not a long-running process)
                let rust_fun = Functional {
                    weights,
                    vars,
                    mode: Mode::PartialDerivatives,
                    order,
                };

                for (point_idx, gp) in grid.iter().enumerate() {
                    let input = build_input(gp, vars);
                    let mut rust_out = vec![0.0_f64; outlen];
                    let mut cpp_out = vec![0.0_f64; outlen];

                    if let Err(e) = rust_fun.eval(&input, &mut rust_out) {
                        anyhow::bail!("Rust eval failed at {} order={} point={}: {:?}", name, order, point_idx, e);
                    }
                    cpp.eval(&input, &mut cpp_out);

                    for elem_idx in 0..outlen {
                        let r = rust_out[elem_idx];
                        let c = cpp_out[elem_idx];
                        let abs_err = (r - c).abs();
                        let rel_err = abs_err / c.abs().max(1.0);
                        let pass = rel_err <= threshold;
                        let rec = ReportRecord {
                            functional: name.into(),
                            vars: format!("{:?}", vars),
                            mode: 1,
                            order,
                            point_idx,
                            element_idx: elem_idx,
                            input: input.clone(),
                            rust: r,
                            cpp: c,
                            abs_err,
                            rel_err,
                            threshold,
                            pass,
                        };
                        report.push(rec);
                    }
                }
            }
        }

        tracing::info!("Tier-2 done: {} records evaluated, {} failed", report.total_records(), report.failed_count());
        Ok(report)
    }
    ```
