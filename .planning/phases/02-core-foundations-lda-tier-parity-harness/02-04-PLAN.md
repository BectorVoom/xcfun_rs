<!-- Mode::PartialDerivatives output layout (XCFunctional.cpp:493-617) -->
For Mode::PartialDerivatives, inlen = input_length(vars), order ∈ {0, 1, 2}:
- Order 0 (outlen = 1): output[0] = energy = d.eval at all-zero higher-coeffs
  Single launch with N=0 (length-1 Array).
- Order 1 (outlen = 1 + inlen): output[0] = energy, output[i+1] = ∂/∂input[i] for i ∈ 0..inlen
  Per input slot i: launch with N=1 (length-2 Array), input[i] gets VAR0=1, others VAR0=0;
  read out[CNST]=energy (only output to output[0] once on the first launch),
  read out[VAR0]=∂/∂input[i] → output[i+1].
- Order 2 (outlen = 1 + inlen + inlen*(inlen+1)/2): output[0] = energy,
  output[1..=inlen] = first derivatives,
  output[inlen+1..] = second derivatives in upper-triangular packed order.
  Per (i, j) pair with i ≤ j: launch with N=2 (length-4 Array), input[i] gets VAR0=1,
  input[j] gets VAR1=1; read out[CNST]=energy, out[VAR0]=∂/∂i, out[VAR1]=∂/∂j,
  out[VAR0|VAR1]=∂²/∂i∂j → output[k++].

    //! - `xcfun-master/src/functionals/slater.hpp:19-21` (formula)

    //! # Preconditions
    //! - `d.a_43`, `d.b_43` populated by `build_densvars` (XC_A_B variant arm — Plan 02-03 Wave-1B-3).

    use cubecl::prelude::*;
    use xcfun_ad::ctaylor::{ctaylor_add, ctaylor_scalar_mul};

    use crate::density_vars::DensVarsDev;

    /// `(81 / (32 * π))^(1/3)` — matches `xcfun-core::constants::C_SLATER`.
    const NEG_C_SLATER_F32: f32 = -0.9305257363491002_f32;

    #[cube]
    pub fn slaterx_kernel<F: Float, const N: u32>(
        d: &DensVarsDev<F>,
        out: &mut Array<F>,
    ) {
        // Operation order (left-to-right; ACC-06 forbids mul_add):
        //   1. tmp = a_43 + b_43         (ctaylor_add)
        //   2. out = (-c_slater) * tmp    (ctaylor_scalar_mul)

        ctaylor_add::<F>(&d.a_43, &d.b_43, &mut tmp, N);
        ctaylor_scalar_mul::<F>(&tmp, F::new(NEG_C_SLATER_F32), out, N);
    }

        }
