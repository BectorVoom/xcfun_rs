    /// CRITICAL per ACC-03: operation order preserved. NO algebraic simplification.
    /// Branch domain: uses `ctaylor_sqrtx_asinh_sqrtx` from D-06 (plan 03-00).
    ///
    /// C++ formula:
    ///   num   = 1 + a1·s²·sqrtx_asinh_sqrtx(a2·s²) + (a3 - a4·exp(-100·s²))·s²
    ///   denom = 1 + a1·s²·sqrtx_asinh_sqrtx(a2·s²) + b·s⁴
    ///   out   = num / denom
    #[cube]
    pub fn pw91xk_enhancement<F: Float>(
        s2: &Array<F>,
        a1: F, a2: F, a3: F, a4: F, _a5: F, b: F,
        out: &mut Array<F>, #[comptime] n: u32,
    ) {
        let size = comptime!((1_u32 << n) as usize);

        // ---- Shared sub-expression: a1·s²·sqrtx_asinh_sqrtx(a2·s²) ----
        // a2_s2 = a2 · s²
        let mut a2_s2 = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(s2, a2, &mut a2_s2, n);

        // sas = sqrtx_asinh_sqrtx(a2_s2)   (D-06 Padé-branched)
        let mut sas = Array::<F>::new(size);
        ctaylor_sqrtx_asinh_sqrtx::<F>(&a2_s2, &mut sas, n);

        // a1_s2 = a1 · s²
        let mut a1_s2 = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(s2, a1, &mut a1_s2, n);

        // shared = a1·s² · sas
        let mut shared = Array::<F>::new(size);
        ctaylor_mul::<F>(&a1_s2, &sas, &mut shared, n);

        // ---- Numerator-only tail: (a3 - a4·exp(-100·s²))·s² ----
        // neg_100_s2 = -100 · s²
        let mut neg_100_s2 = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(s2, F::cast_from(-100.0_f64), &mut neg_100_s2, n);

        // exp_neg_100_s2 = exp(-100·s²)
        let mut exp_neg_100_s2 = Array::<F>::new(size);
        ctaylor_exp::<F>(&neg_100_s2, &mut exp_neg_100_s2, n);

        // a4_exp = a4 · exp(-100·s²)
        let mut a4_exp = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(&exp_neg_100_s2, a4, &mut a4_exp, n);

        // a3_ct = CNST(a3)  (CNST-scratch array to enable ctaylor_sub)
        let mut a3_ct = Array::<F>::new(size);
        #[unroll] for i in 0_u32..(1_u32 << n) { a3_ct[i as usize] = F::new(0.0); }
        a3_ct[0] = a3;

        // bracket = a3 - a4·exp(-100·s²)
        let mut bracket = Array::<F>::new(size);
        ctaylor_sub::<F>(&a3_ct, &a4_exp, &mut bracket, n);

        // num_tail = bracket · s²
        let mut num_tail = Array::<F>::new(size);
        ctaylor_mul::<F>(&bracket, s2, &mut num_tail, n);

        // ---- Denominator-only tail: b·s⁴ ----
        // s4 = s² · s²
        let mut s4 = Array::<F>::new(size);
        ctaylor_mul::<F>(s2, s2, &mut s4, n);

        // den_tail = b · s⁴
        let mut den_tail = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(&s4, b, &mut den_tail, n);

        // ---- Assemble numerator = 1 + shared + num_tail ----
        let mut one_ct = Array::<F>::new(size);
        #[unroll] for i in 0_u32..(1_u32 << n) { one_ct[i as usize] = F::new(0.0); }
        one_ct[0] = F::new(1.0);
        let mut num_partial = Array::<F>::new(size);
        ctaylor_add::<F>(&one_ct, &shared, &mut num_partial, n);
        let mut num = Array::<F>::new(size);
        ctaylor_add::<F>(&num_partial, &num_tail, &mut num, n);

        // ---- Assemble denominator = 1 + shared + den_tail ----
        let mut den_partial = Array::<F>::new(size);
        ctaylor_add::<F>(&one_ct, &shared, &mut den_partial, n);
        let mut den = Array::<F>::new(size);
        ctaylor_add::<F>(&den_partial, &den_tail, &mut den, n);

        // ---- out = num / den ----
        let mut inv_den = Array::<F>::new(size);
        ctaylor_reciprocal::<F>(&den, &mut inv_den, n);
        ctaylor_mul::<F>(&num, &inv_den, out, n);
    }
