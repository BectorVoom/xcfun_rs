    //! # Source: xcfun-master/src/functionals/beckex.cpp:17-25, 75-77
    //! # Formula (single-spin):
    //!   chi² = gaa / a^(8/3) (chi² helper via pw91_like::chi2(a, gaa))
    //!   sas = sqrt(chi²) · asinh(sqrt(chi²))   (via ctaylor_sqrtx_asinh_sqrtx(chi²))
    //!   denom = 1 + 6·d·sas                    (BECKE_6D_F64 = 0.0252)
    //!   num   = d · chi²                       (BECKE_D_F64   = 0.0042)
    //!   E_x_alpha = -C_SLATER · a^(4/3) · (1 + num/denom)
    //!   return E_x_alpha + E_x_beta
    use cubecl::prelude::*;
    use xcfun_ad::ctaylor::{ctaylor_add, ctaylor_scalar_mul, ctaylor_sub};
    use xcfun_ad::ctaylor_rec::mul::ctaylor_mul;
    use xcfun_ad::math::{ctaylor_sqrtx_asinh_sqrtx, ctaylor_reciprocal};
    use crate::density_vars::DensVarsDev;
    use crate::functionals::gga::shared::{pw91_like, constants::*};

    #[cube]
    pub fn beckex_kernel<F: Float>(
        d: &DensVarsDev<F>, out: &mut Array<F>, #[comptime] n: u32,
    ) {
        let size = comptime!((1_u32 << n) as usize);

        // --- alpha-channel ---
        let mut chi2_a  = Array::<F>::new(size);
        let mut sas_a   = Array::<F>::new(size);
        let mut six_d_sas_a = Array::<F>::new(size);
        let mut denom_a = Array::<F>::new(size);
        let mut inv_denom_a = Array::<F>::new(size);
        let mut num_a   = Array::<F>::new(size);
        let mut ratio_a = Array::<F>::new(size);
        let mut one_plus_a = Array::<F>::new(size);
        let mut one_ct_a = Array::<F>::new(size);
        let mut lsda_a  = Array::<F>::new(size);
        let mut e_alpha = Array::<F>::new(size);

        pw91_like::chi2::<F>(&d.a, &d.gaa, &mut chi2_a, n);
        ctaylor_sqrtx_asinh_sqrtx::<F>(&chi2_a, &mut sas_a, n);
        ctaylor_scalar_mul::<F>(&sas_a, F::cast_from(BECKE_6D_F64), &mut six_d_sas_a, n);

        // denom = 1 + 6·d·sas via CNST-scratch add
        #[unroll] for i in 0_u32..(1_u32 << n) { one_ct_a[i as usize] = F::new(0.0); }
        one_ct_a[0] = F::new(1.0);
        ctaylor_add::<F>(&one_ct_a, &six_d_sas_a, &mut denom_a, n);
        ctaylor_reciprocal::<F>(&denom_a, &mut inv_denom_a, n);

        // num = d · chi²
        ctaylor_scalar_mul::<F>(&chi2_a, F::cast_from(BECKE_D_F64), &mut num_a, n);

        // ratio = num · inv_denom
        ctaylor_mul::<F>(&num_a, &inv_denom_a, &mut ratio_a, n);

        // one_plus = 1 + ratio
        ctaylor_add::<F>(&one_ct_a, &ratio_a, &mut one_plus_a, n);

        // lsda = -C_SLATER · a_43
        ctaylor_scalar_mul::<F>(&d.a_43, F::cast_from(NEG_C_SLATER_F64), &mut lsda_a, n);

        // e_alpha = lsda · one_plus
        ctaylor_mul::<F>(&lsda_a, &one_plus_a, &mut e_alpha, n);

        // --- beta-channel (mirror) ---
        let mut chi2_b = Array::<F>::new(size);
        let mut sas_b  = Array::<F>::new(size);
        let mut six_d_sas_b = Array::<F>::new(size);
        let mut denom_b = Array::<F>::new(size);
        let mut inv_denom_b = Array::<F>::new(size);
        let mut num_b   = Array::<F>::new(size);
        let mut ratio_b = Array::<F>::new(size);
        let mut one_plus_b = Array::<F>::new(size);
        let mut one_ct_b = Array::<F>::new(size);
        let mut lsda_b  = Array::<F>::new(size);
        let mut e_beta  = Array::<F>::new(size);

        pw91_like::chi2::<F>(&d.b, &d.gbb, &mut chi2_b, n);
        ctaylor_sqrtx_asinh_sqrtx::<F>(&chi2_b, &mut sas_b, n);
        ctaylor_scalar_mul::<F>(&sas_b, F::cast_from(BECKE_6D_F64), &mut six_d_sas_b, n);
        #[unroll] for i in 0_u32..(1_u32 << n) { one_ct_b[i as usize] = F::new(0.0); }
        one_ct_b[0] = F::new(1.0);
        ctaylor_add::<F>(&one_ct_b, &six_d_sas_b, &mut denom_b, n);
        ctaylor_reciprocal::<F>(&denom_b, &mut inv_denom_b, n);
        ctaylor_scalar_mul::<F>(&chi2_b, F::cast_from(BECKE_D_F64), &mut num_b, n);
        ctaylor_mul::<F>(&num_b, &inv_denom_b, &mut ratio_b, n);
        ctaylor_add::<F>(&one_ct_b, &ratio_b, &mut one_plus_b, n);
        ctaylor_scalar_mul::<F>(&d.b_43, F::cast_from(NEG_C_SLATER_F64), &mut lsda_b, n);
        ctaylor_mul::<F>(&lsda_b, &one_plus_b, &mut e_beta, n);

        // out = e_alpha + e_beta
        ctaylor_add::<F>(&e_alpha, &e_beta, out, n);
    }
