    use xcfun_ad::ctaylor::{ctaylor_add, ctaylor_scalar_mul};
    use xcfun_ad::ctaylor_rec::mul::ctaylor_mul;
    use xcfun_ad::math::ctaylor_reciprocal;

    #[cube]
    pub fn ux_ab<F: Float>(
        gamma: F, s2: &Array<F>,
        out: &mut Array<F>, #[comptime] n: u32,
    ) {
        let size = comptime!((1_u32 << n) as usize);
        // num_term = γ · s²
        let mut num_term = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(s2, gamma, &mut num_term, n);
        // denom = 1 + γ·s²  (via CNST scratch then add)
        let mut one_ct = Array::<F>::new(size);
        #[unroll] for i in 0_u32..(1_u32 << n) { one_ct[i as usize] = F::new(0.0); }
        one_ct[0] = F::new(1.0);
        let mut denom = Array::<F>::new(size);
        ctaylor_add::<F>(&one_ct, &num_term, &mut denom, n);
        // inv_denom
        let mut inv_denom = Array::<F>::new(size);
        ctaylor_reciprocal::<F>(&denom, &mut inv_denom, n);
        // out = num_term · inv_denom
        ctaylor_mul::<F>(&num_term, &inv_denom, out, n);
    }

    **Step B — `shared/b97_poly.rs::b97_enhancement` FULL BODY (Pitfall G6 verbatim):**
    ```rust
    /// Degree-2 B97 enhancement: c0 + c1·u + c2·u²  (b97xc.hpp)
    /// PITFALL G6: preserve operator precedence — NO Horner form.
    #[cube]
    pub fn b97_enhancement<F: Float>(
        c0: F, c1: F, c2: F, u: &Array<F>,
        out: &mut Array<F>, #[comptime] n: u32,
    ) {
        let size = comptime!((1_u32 << n) as usize);
        // u² = u · u   (explicit — do NOT fuse with c1·u into a mul_add)
        let mut u2 = Array::<F>::new(size);
        ctaylor_mul::<F>(u, u, &mut u2, n);
        // term1 = c1 · u
        let mut term1 = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(u, c1, &mut term1, n);
        // term2 = c2 · u²
        let mut term2 = Array::<F>::new(size);
        ctaylor_scalar_mul::<F>(&u2, c2, &mut term2, n);
        // sum = term1 + term2
        let mut sum = Array::<F>::new(size);
        ctaylor_add::<F>(&term1, &term2, &mut sum, n);
        // Add CNST c0 via scratch
        let mut c0_ct = Array::<F>::new(size);
        #[unroll] for i in 0_u32..(1_u32 << n) { c0_ct[i as usize] = F::new(0.0); }
        c0_ct[0] = c0;
        ctaylor_add::<F>(&c0_ct, &sum, out, n);
    }
    ```

        d: &DensVarsDev<F>, out: &mut Array<F>, #[comptime] n: u32,

        let size = comptime!((1_u32 << n) as usize);
