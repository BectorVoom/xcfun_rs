template <class T, int Nvar>
static ctaylor<T, Nvar> expm1(const ctaylor<T, Nvar> & t) {
  T tmp[Nvar + 1];
  exp_expand<T, Nvar>(tmp, t.c[0]);
  if (fabs(t.c[0]) > 1e-3)
    tmp[0] -= 1;
  else
    tmp[0] = 2 * exp(t.c[0] / 2) * sinh(t.c[0] / 2);
  ctaylor<T, Nvar> res;
  ctaylor_rec<T, Nvar>::compose(res.c, t.c, tmp);
  return res;
}

    /// Port target: xcfun-master/external/upstream/taylor/ctaylor_math.hpp:275-325.
    /// Precondition: x[0] >= 0. |x[0]| >= 0.5 uses direct composition;
    /// |x[0]| < 0.5 uses [8,8] Padé approximant to preserve 1e-14 precision at x → 0.
    #[cube]
    pub fn ctaylor_sqrtx_asinh_sqrtx<F: Float>(
        x: &Array<F>, out: &mut Array<F>, #[comptime] n: u32,
    ) {
        let size = comptime!((1_u32 << n) as usize);
        // Branch threshold per upstream.
        let threshold = F::cast_from(0.5_f64);
        let abs_x0 = x[0].abs();

        if abs_x0 >= threshold {
            // UNSTABLE branch — direct composition is numerically fine for x0 >= 0.5.
            let mut sx = Array::<F>::new(size);
            let mut asx = Array::<F>::new(size);
            ctaylor_sqrt::<F>(x, &mut sx, n);
            ctaylor_asinh::<F>(&sx, &mut asx, n);

            // evaluated at the origin (y = x), then shifted to x0.
