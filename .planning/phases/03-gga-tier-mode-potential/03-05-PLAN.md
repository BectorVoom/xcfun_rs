            if deps.contains(Dependency::LAPLACIAN) || deps.contains(Dependency::KINETIC) {

            // Inlen + nspin computation (mirrors XCFunctional.cpp:639-652):

            let (npot, inpos) = match inlen {
                1 | 10 => (1_usize, 0_usize),   // nspin=1
                2      => (2, 1),
                20     => (2, 10),

            //   coeff[CNST] = input[l]; coeff[VAR0] = 0. Then set coeff[VAR0] = 1 on

            let mut ct_in = vec![0.0_f64; inlen * SIZE_N1];   // flat CTaylor block

                // Populate CNST and zero VAR0 everywhere.
                for l in 0..inlen {
                    ct_in[l * SIZE_N1 + 0 /* CNST */] = input[l];
                    ct_in[l * SIZE_N1 + 1 /* VAR0 */] = 0.0;

    fn launch_potential_gga(&self, input: &[f64], out: &mut [f64]) -> Result<(), XcError> {
        const SIZE_N2: usize = 4;   // 1 << 2

        // Helper: populate a CTaylor<f64, 2> block for one direction, one spin slot.
        //
        // `inlen` = 10 (nspin=1) or 20 (nspin=2).
        //

        //   in[0 + offset].CNST = input[0 + offset]
        //   in[0 + offset].VAR0 = input[(1 + dir) + offset]          (density gradient component)
        //   in[1 + offset].CNST = input[1 + offset]
        //   in[1 + offset].VAR0 = input[hess_slot(1, dir) + offset]
        //   in[2 + offset].CNST = input[2 + offset]
        //   in[2 + offset].VAR0 = input[hess_slot(2, dir) + offset]
        //   in[3 + offset].CNST = input[3 + offset]
        //   in[3 + offset].VAR0 = input[hess_slot(3, dir) + offset]
        //   (slots 4..9, 14..19: all zero per XCFunctional.cpp:687 + :745)
        // Plus: VAR1=1 on in[(1 + dir) + offset * j] where j is the spin-direction flag.
        //
        // The hess_slot table comes directly from XCFunctional.cpp:683-713:
        //   hess_slot(src, dir):
        //     dir=x:  src=1→4,  src=2→5,  src=3→6
        //     dir=y:  src=1→5,  src=2→7,  src=3→8
        //     dir=z:  src=1→6,  src=2→8,  src=3→9
        const HESS_SLOT: [[usize; 3]; 3] = [
            /* src=1 */ [4, 5, 6],
            /* src=2 */ [5, 7, 8],
            /* src=3 */ [6, 8, 9],
        ];

        let inlen = Self::input_length(self.vars);
        let nspin = match inlen {
            10 => 1_usize,
            20 => 2,
            _  => return Err(XcError::InvalidVars { vars: self.vars }),
        };

        let mut ct_in = vec![0.0_f64; inlen * SIZE_N2];
        let mut energy_accum = 0.0_f64;
        let mut pot_accum = vec![0.0_f64; nspin]; // one accumulator per spin channel

        for j in 0..nspin {
            let offset = if nspin == 2 { 10_usize } else { 0 };
            let active_offset = offset * j;   // 0 for α, 10 for β

            // j-specific accumulator for out[VAR0|VAR1] contributions across 3 dirs.
            let mut divergence_accum = 0.0_f64;

                // Zero everything first, then overwrite.

                // only the VAR1 seed discriminates which channel's divergence we're computing.
                let spin_offsets: &[usize] = if nspin == 2 { &[0, 10] } else { &[0] };
                for &off in spin_offsets {
                    // in[0 + off].CNST = input[0 + off]
                    ct_in[(0 + off) * SIZE_N2 + 0] = input[0 + off];
                    // in[0 + off].VAR0 = input[(1 + dir) + off]   (first-order density gradient along dir)
                    ct_in[(0 + off) * SIZE_N2 + 1] = input[(1 + dir) + off];

                    // in[1..=3 + off] populated from HESS_SLOT table.
                    for src in 1_usize..=3 {
                        ct_in[(src + off) * SIZE_N2 + 0] = input[src + off];
                        ct_in[(src + off) * SIZE_N2 + 1] = input[HESS_SLOT[src - 1][dir] + off];
                    }
                    // slots 4..9 (and 14..19 for nspin=2) already zero per memset above.
                }

                // Slot index: (1 + dir) + active_offset.

                ct_in[var1_slot * SIZE_N2 + 2 /* VAR1 */] = 1.0;

                // ----- Launch potential_gga_kernel for each active (id, weight) -----
                let mut kernel_out = vec![0.0_f64; SIZE_N2];
                for &(id, w) in self.weights {
                    self.launch_potential_kernel_n2(id as u32, &ct_in, &mut kernel_out)?;
                    // Energy: only accumulate once (j==0, dir==0).
                    if j == 0 && dir == 0 {
                        energy_j += w * kernel_out[0 /* CNST */];
                    }
                    // Divergence-integrand contribution:
                    //   out[VAR0|VAR1] (slot 3 = 0b11).
                    divergence_accum += w * kernel_out[3];

                    // For nspin=1, the LDA-direct potential term (∂E/∂ρ) is captured once
                    // as well — XCFunctional.cpp:666 + :721 implies the -∇·(∂E/∂g) subtracts
                    // FROM the LDA-direct pot term already populated by Mode::Potential LDA dispatch.
                    // Executor MUST verify: for GGA, the LDA direct term arrives via kernel_out[VAR0]
                    // at dir==0 only (or is folded into the full ctaylor "out" as in XCFunctional.cpp).
                    // Preserve operation order per ACC-03 — do NOT combine these two accumulators early.
                }
            }

            if j == 0 { energy_accum = energy_j; }

            //   output[j + 1] -= out.get(VAR0 | VAR1);     (subtract divergence)
            //
            // The `-=` implies output[j+1] is already populated with the LDA-style
            // direct potential ∂E/∂ρ before this subtraction. For our single-pass
            // design, executor MUST populate out[j+1] = weighted LDA direct pot
            // (from kernel_out[VAR0] accumulated above) FIRST, then subtract
            // divergence_accum. Preserve the subtract-from-populated invariant.
            pot_accum[j] = /* weighted LDA pot Σ w_i·out[VAR0] */ 0.0 - divergence_accum;

            // OR, more faithfully, via a separate N=1 re-launch mirroring launch_potential_lda;
            // verify against C++ reference at Task 3 integration test).
        }

        out[0] = energy_accum;
        for j in 0..nspin { out[j + 1] = pot_accum[j]; }
        Ok(())
    }
