        /// Line-for-line port of XCFunctional.cpp:637-790 (D-13).
        /// Two-pass structure per XCFunctional.cpp:671 (the GGA block runs AFTER
        /// the LDA loop and uses `-=` on the already-populated output slots).
        pub fn launch_potential(&self, input: &[f64], out: &mut [f64]) -> Result<(), XcError> {
            let deps = self.dependencies();

            // Defense-in-depth (eval_setup from plan 03-01 already enforced these).
            if deps.contains(Dependency::LAPLACIAN) || deps.contains(Dependency::KINETIC) {
                return Err(XcError::InvalidMode { mode: self.mode, depends: deps });
            }

            // Pass 1 (XCFunctional.cpp:653-670): ALWAYS run the LDA N=1 loop first.
            // Populates out[0] = energy, out[j+1] = ∂E/∂ρ (LDA-direct term).
            self.launch_potential_lda(input, out)?;

            // Pass 2 (XCFunctional.cpp:671-791): if GRADIENT, subtract divergence.
            if deps.contains(Dependency::GRADIENT) {
                self.launch_potential_gga(input, out)?;
            }
            Ok(())
        }

        /// Port of XCFunctional.cpp:637-670 — LDA path at N=1.
        /// ALWAYS runs (even for GGA functionals) to populate the LDA-direct
        /// potential term before the GGA block subtracts divergence.
        fn launch_potential_lda(&self, input: &[f64], out: &mut [f64]) -> Result<(), XcError> {
            // Inlen + nspin computation (mirrors XCFunctional.cpp:639-652):
            let inlen = Self::input_length(self.vars);
            let (npot, inpos) = match inlen {
                1 | 10 => (1_usize, 0_usize),   // nspin=1
                2      => (2, 1),
                20     => (2, 10),
                _      => return Err(XcError::InvalidVars { vars: self.vars }),
            };

            // Size at N=1 is 2 (CNST + VAR0).
            const SIZE_N1: usize = 2;

            // For j = 0 (α) then optionally j = 1 (β):
            //   Pre-seed a CTaylor<f64, 1> input block: for every slot l in 0..inlen,
            //   coeff[CNST] = input[l]; coeff[VAR0] = 0. Then set coeff[VAR0] = 1 on
            //   the density slot (l == j * inpos).
            let mut ct_in = vec![0.0_f64; inlen * SIZE_N1];   // flat CTaylor block
            let mut energy_accum = 0.0_f64;

            for j in 0..npot {
                // Populate CNST and zero VAR0 everywhere.
                for l in 0..inlen {
                    ct_in[l * SIZE_N1 + 0 /* CNST */] = input[l];
                    ct_in[l * SIZE_N1 + 1 /* VAR0 */] = 0.0;
                }
                // Seed VAR0 = 1 on density slot (l = j * inpos per XCFunctional.cpp:659).
                ct_in[(j * inpos) * SIZE_N1 + 1] = 1.0;

                // Launch dispatch_kernel at N=1 for each active (id, weight) in self.weights,
                // accumulating weighted out[CNST] (energy) + out[VAR0] (potential).
                let mut kernel_out = vec![0.0_f64; SIZE_N1];
                let mut weighted_energy = 0.0_f64;
                let mut weighted_pot    = 0.0_f64;
                for &(id, w) in self.weights {
                    self.launch_potential_kernel_n1(id as u32, &ct_in, &mut kernel_out)?;
                    weighted_energy += w * kernel_out[0];
                    weighted_pot    += w * kernel_out[1];
                }

                // Output slot j+1 receives the LDA-direct potential (XCFunctional.cpp:666).
                // NOTE: for GGA functionals, launch_potential_gga will subtract divergence
                //       from this populated slot via `pot_accum[j] -= divergence`.
                out[j + 1] = weighted_pot;
                // Energy accumulates only once (same value across j for LDA).
                if j == 0 { energy_accum = weighted_energy; }
            }

            // Energy at out[0] (XCFunctional.cpp:669).
            out[0] = energy_accum;
            Ok(())
        }

        /// Build flat ct_in + launch potential_lda_kernel at N=1.
        /// Body mirrors `run_launch` + `launch_eval_point` at
        /// crates/xcfun-eval/src/functional.rs:288-484, specialised to N=1.
        /// EXECUTOR: this is a RENAMED/SPECIALISED view of existing Phase-2
        /// infrastructure — copy the `run_launch` body verbatim with N=1
        /// threaded through as a const, then dispatch on (id, 1) instead of
        /// (id, n). Do NOT leave the body as `Ok(())`.
        #[cfg(feature = "testing")]
        fn launch_potential_kernel_n1(
            &self, id: u32, ct_in: &[f64], kernel_out: &mut [f64],
        ) -> Result<(), XcError> {
            // Inline port of run_launch at functional.rs:288-433 with n=1.
            // We reuse the existing `launch_eval_point::<ID, VARS, N>` wrapper
            // at functional.rs:438-484 — `launch_potential_kernel_n1` is
            // nothing but `run_launch(id, self.vars as u32, 1, ct_in, 2)`.
            const OUT_LEN_N1: usize = 2;   // 1 << 1
