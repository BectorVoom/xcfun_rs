**test_all_lda_functionals** (per D-19, VAL-02):
```rust
use xcfun_core::test_data::LDA_TEST_CASES;
use xcfun_core::{taylorlen, FunctionalId};
use xcfun_eval::XcFunctional;

#[test]
fn test_all_lda_functionals() {
    for tc in LDA_TEST_CASES {
        let name = tc.id.name();
        let mut xc = XcFunctional::new();
        xc.set(name, 1.0).unwrap_or_else(|e| panic!("{name}: set failed: {e}"));
        xc.eval_setup(tc.vars, tc.mode, tc.order)
            .unwrap_or_else(|e| panic!("{name}: eval_setup failed: {e}"));

        let output_len = taylorlen(tc.vars.input_len(), tc.order as usize);
        let mut output = vec![0.0f64; output_len];
        xc.eval(tc.input, &mut output)
            .unwrap_or_else(|e| panic!("{name}: eval failed: {e}"));

        // Accuracy reporting (per D-20, VAL-03)
        let mut max_rel_err = 0.0f64;
        let mut sum_rel_err = 0.0f64;
        let mut n_compared = 0usize;

        for (i, (&got, &expected)) in output.iter().zip(tc.expected.iter()).enumerate() {
            if expected.abs() < 1e-30 {
                // Near-zero: use absolute comparison
                let abs_err = (got - expected).abs();
                if abs_err > tc.threshold {
                    panic!(
                        "{name}: output[{i}] absolute error {abs_err:.3e} exceeds threshold {:.3e}\n  got:      {got:.15e}\n  expected: {expected:.15e}",
                        tc.threshold
                    );
                }
                continue;
            }
            let rel_err = ((got - expected) / expected).abs();
            max_rel_err = max_rel_err.max(rel_err);
            sum_rel_err += rel_err;
            n_compared += 1;

            if rel_err > tc.threshold {
                let mean_so_far = if n_compared > 0 { sum_rel_err / n_compared as f64 } else { 0.0 };
                panic!(
                    "{name}: output[{i}] relative error {rel_err:.3e} exceeds threshold {:.3e}\n  got:      {got:.15e}\n  expected: {expected:.15e}\n  max_err:  {max_rel_err:.3e}\n  mean_err: {mean_so_far:.3e}",
                    tc.threshold
                );
            }
        }

        let mean_rel_err = if n_compared > 0 { sum_rel_err / n_compared as f64 } else { 0.0 };
        eprintln!("{name}: PASS (max_rel_err={max_rel_err:.3e}, mean_rel_err={mean_rel_err:.3e}, n={n_compared})");
    }
}
```

**test_lda_aliases** (per LDA-06 cross-validation):

```rust
#[test]
fn test_lda_alias_svwn5() {
    // Evaluate svwn5 alias
    let mut xc_alias = XcFunctional::new();
    xc_alias.set("svwn5", 1.0).unwrap();
    xc_alias.eval_setup(VarType::A_B, EvalMode::PartialDerivatives, 2).unwrap();
    let mut out_alias = vec![0.0; taylorlen(2, 2)];
    xc_alias.eval(&[39.0, 38.0], &mut out_alias).unwrap();

    // Evaluate components separately
    let mut xc_parts = XcFunctional::new();
    xc_parts.set("slaterx", 1.0).unwrap();
    xc_parts.set("vwn5c", 1.0).unwrap();
    xc_parts.eval_setup(VarType::A_B, EvalMode::PartialDerivatives, 2).unwrap();
    let mut out_parts = vec![0.0; taylorlen(2, 2)];
    xc_parts.eval(&[39.0, 38.0], &mut out_parts).unwrap();

    for (i, (&a, &p)) in out_alias.iter().zip(out_parts.iter()).enumerate() {
        assert!(
            (a - p).abs() < 1e-14,
            "svwn5 alias vs components differ at [{i}]: {a:.15e} vs {p:.15e}"
        );
    }
}
```

**test_cross_validate_orders** (per VAL-04):
For each LDA functional with test data, evaluate at orders 0, 1, 2 and verify:
- Order 0: output has 1 element (energy), matches expected[0]
- Order 1: output has input_len+1 elements, matches expected[0..input_len+1]
- Order 2: output matches full expected array (the test data is at order 2)
```rust
#[test]
fn test_cross_validate_orders() {
    for tc in LDA_TEST_CASES {
        let name = tc.id.name();
        for order in 0..=tc.order.min(4) {
            let mut xc = XcFunctional::new();
            xc.set(name, 1.0).unwrap();
            xc.eval_setup(tc.vars, tc.mode, order).unwrap();

            let n_vars = tc.vars.input_len();
            let out_len = taylorlen(n_vars, order as usize);
            let mut output = vec![0.0; out_len];
            xc.eval(tc.input, &mut output).unwrap();

            // Verify against the first out_len elements of expected
            // (lower order is a subset of higher order output)
            for (i, &got) in output.iter().enumerate() {
                if i >= tc.expected.len() { break; }
                let expected = tc.expected[i];
                if expected.abs() < 1e-30 { continue; }
                let rel_err = ((got - expected) / expected).abs();
                assert!(
                    rel_err < tc.threshold * 10.0, // slightly relaxed for subset extraction
                    "{name} order {order} [{i}]: rel_err {rel_err:.3e} (got {got:.15e}, exp {expected:.15e})"
                );
            }
        }
    }
}
```
