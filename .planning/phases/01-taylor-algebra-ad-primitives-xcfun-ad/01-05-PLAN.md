    ```rust
    impl<const N: usize> Num for CTaylor<f64, N>
    where
        Bound: ValidN<N>,
        // Required trait bounds for Mul/Div/compose on CTaylor from Plan 03.
        [f64; 1 << N]: crate::ctaylor_rec::MulDispatch + crate::ctaylor_rec::ComposeDispatch,
    {
        const ZERO: Self = Self { c: [0.0_f64; 1 << N] };
        const ONE: Self = {
            let mut c = [0.0_f64; 1 << N];
            c[0] = 1.0;
            Self { c }
        };

        fn reciprocal(self) -> Self { crate::math::ctaylor_reciprocal(self) }
        fn sqrt(self) -> Self { crate::math::ctaylor_sqrt(self) }
        fn exp(self) -> Self { crate::math::ctaylor_exp(self) }
        fn log(self) -> Self { crate::math::ctaylor_log(self) }
        fn pow(self, a: Self) -> Self {
            // ctaylor_math.hpp:117-131 — the scalar `a` path; if `a` is not purely scalar,
            // fall back to `exp(a * log(self))`.
            // W11 revision: the non-scalar branch has NO byte-equivalence contract against C++
            // (C++ xcfun has no `pow(ctaylor, ctaylor)`). Golden fixtures only cover the scalar
            // branch. See module-header comment in math.rs.
            if is_scalar(&a) {
                crate::math::ctaylor_pow(self, a.c[0])
            } else {
                crate::math::ctaylor_exp(a * crate::math::ctaylor_log(self))
            }
        }
        fn powi(self, n: i32) -> Self { crate::math::ctaylor_powi(self, n) }
        fn erf(self) -> Self { crate::math::ctaylor_erf(self) }
        fn asinh(self) -> Self { crate::math::ctaylor_asinh(self) }
        fn atan(self) -> Self { crate::math::ctaylor_atan(self) }
    }

    fn is_scalar<const N: usize>(t: &CTaylor<f64, N>) -> bool
    where Bound: ValidN<N>
    {
        t.c[1..].iter().all(|&x| x == 0.0)
    }
    ```

    Implementation:
    ```rust
    use crate::{CTaylor, CNST};
    use crate::valid_n::{Bound, ValidN};
    use crate::expand;
    use crate::ctaylor_rec::{self, ComposeDispatch, MulDispatch};

    const N_MAX_PLUS_1: usize = 8;   // N_MAX = 7; scratch size = 8

    /// `1 / t` via inv_expand + compose. ctaylor_math.hpp:7-28.
    /// Precondition: `t.c[CNST] != 0`.
    pub fn ctaylor_reciprocal<const N: usize>(t: CTaylor<f64, N>) -> CTaylor<f64, N>
    where
        Bound: ValidN<N>,
        [f64; 1 << N]: MulDispatch + ComposeDispatch,
    {
        let mut scratch = [0.0_f64; N_MAX_PLUS_1];
        let slice = &mut scratch[..=N];
        expand::inv_expand(slice, t.c[CNST]);

        let mut result = CTaylor::<f64, N>::from_scalar(0.0);
        // B2 revision: dispatch locked to ComposeDispatch trait — no 'executor chooses'.
        <[f64; 1 << N] as ComposeDispatch>::compose(&mut result.c, &t.c, slice);
        result
    }

    /// `exp(t)` via exp_expand + compose. ctaylor_math.hpp:71-81.
    pub fn ctaylor_exp<const N: usize>(t: CTaylor<f64, N>) -> CTaylor<f64, N>
    where
        Bound: ValidN<N>,
        [f64; 1 << N]: MulDispatch + ComposeDispatch,
    {
        let mut scratch = [0.0_f64; N_MAX_PLUS_1];
        let slice = &mut scratch[..=N];
        expand::exp_expand(slice, t.c[CNST]);

        let mut result = CTaylor::<f64, N>::from_scalar(0.0);
        <[f64; 1 << N] as ComposeDispatch>::compose(&mut result.c, &t.c, slice);
        result
    }

    // … same pattern for log, sqrt, erf, atan, asinh. Each cites its ctaylor_math.hpp line range.

    /// `pow(t, a)` with scalar `a`. ctaylor_math.hpp:117-131.
    /// Precondition: `t.c[CNST] > 0`.
    pub fn ctaylor_pow<const N: usize>(t: CTaylor<f64, N>, a: f64) -> CTaylor<f64, N>
    where
        Bound: ValidN<N>,
        [f64; 1 << N]: MulDispatch + ComposeDispatch,
    {
        let mut scratch = [0.0_f64; N_MAX_PLUS_1];
        let slice = &mut scratch[..=N];
        expand::pow_expand(slice, t.c[CNST], a);

        let mut result = CTaylor::<f64, N>::from_scalar(0.0);
        <[f64; 1 << N] as ComposeDispatch>::compose(&mut result.c, &t.c, slice);
        result
    }

    /// `powi(t, n)` integer exponent fast path. ctaylor_math.hpp:165-178.
    pub fn ctaylor_powi<const N: usize>(t: CTaylor<f64, N>, n: i32) -> CTaylor<f64, N>
    where
        Bound: ValidN<N>,
        [f64; 1 << N]: MulDispatch + ComposeDispatch,
    {
        match n.cmp(&0) {
            core::cmp::Ordering::Equal => <CTaylor<f64, N> as crate::Num>::ONE,
            core::cmp::Ordering::Greater => {
                // Loop n times: res = t; res *= t; ... (n-1 multiplies)
                let mut res = t;
                for _ in 1..n {
                    res = res * t;
                }
                res
            }
            core::cmp::Ordering::Less => {
                // n < 0: defer to pow(t, n as f64).
                ctaylor_pow(t, n as f64)
            }
        }
    }
    ```
