```rust
// (tests import via path-dep: xtask = { path = "../../xtask" } is avoided;
//  the struct is duplicated at the xcfun-ad tests/ directory to keep xcfun-ad's

#[derive(serde::Serialize, serde::Deserialize, Debug, Clone)]
pub struct FixtureRecord {
    pub op: String,           // e.g., "mul", "inv_expand", "exp_expand"
    pub n_var: u8,            // 0..=3 for to_bits gate, 4..=6 for rel-err gate
    pub inputs: Vec<f64>,     // flat layout; semantics per-op
    pub coeffs: Vec<f64>,     // expected output coefficient array
}

#[derive(serde::Serialize, serde::Deserialize, Debug)]
pub struct FixturesManifest {
    pub xcfun_version_git_sha: String,      // sha256 of the three taylor .hpp files
    pub generated_at: String,               // RFC 3339 timestamp
    pub per_op_counts: std::collections::BTreeMap<String, usize>,
    pub total_records: usize,
    pub driver_commit: Option<String>,      // the xcfun_rs repo HEAD at regen time
}
```

**Fixture layout by `op`:**
- `op = "mul"`: `inputs` = first `1 << n_var` entries are `a.c[..]`, next `1 << n_var` entries are `b.c[..]`. `coeffs` = `(a * b).c[..]`, length `1 << n_var`.
- `op = "inv_expand"`: `inputs = [a]` (length 1); `coeffs = t[0..=n_var]`, length `n_var + 1`. (Here `n_var` reuses the u8 field for the expansion order; slightly abusing the schema but keeps one struct.)
- `op = "exp_expand"`: `inputs = [x0]`; `coeffs = t[0..=n_var]`.
- `op = "log_expand"`: `inputs = [x0]`; `coeffs = t[0..=n_var]`.
- `op = "pow_expand"`: `inputs = [x0, a]` (length 2); `coeffs = t[0..=n_var]`.
- `op = "sqrt_expand"`, `"cbrt_expand"`, `"gauss_expand"`, `"erf_expand"`: `inputs = [x0]`; `coeffs = t[0..=n_var]`.

    ```cpp
    // via cc::Build. Emits deterministic records to stdout in a line-oriented
    // format that the Rust side parses into Vec<FixtureRecord>.
    //
    // Record format (one per line, semicolon-separated fields):
    //   <op>;<n_var>;<input_count>;<i0>,<i1>,...,<i{k-1}>;<coeff_count>;<c0>,<c1>,...,<c{m-1}>
    //
    // f64 values are written via printf("%.17g", v) to preserve round-trip identity.
    //
    // Output is written to stdout; xtask binary captures it and emits bincode.

    #include <cstdio>
    #include <cstdlib>
    #include <random>
    #include <vector>
    #include "ctaylor.hpp"
    #include "ctaylor_math.hpp"
    #include "tmath.hpp"

    static void emit_record(const char* op, int n_var,
                            const std::vector<double>& inputs,
                            const std::vector<double>& coeffs) {
        printf("%s;%d;%zu;", op, n_var, inputs.size());
        for (size_t i = 0; i < inputs.size(); i++) {
            printf("%.17g", inputs[i]);
            if (i + 1 < inputs.size()) printf(",");
        }
        printf(";%zu;", coeffs.size());
        for (size_t i = 0; i < coeffs.size(); i++) {
            printf("%.17g", coeffs[i]);
            if (i + 1 < coeffs.size()) printf(",");
        }
        printf("\n");
    }

    // --- *_expand record emitters ---

    template <int N>
    static void emit_inv_expand(double a) {
        double t[N + 1];
        inv_expand<double, N>(t, a);
        std::vector<double> inputs = {a};
        std::vector<double> coeffs(t, t + N + 1);
        emit_record("inv_expand", N, inputs, coeffs);
    }

    template <int N>
    static void emit_exp_expand(double x0) {
        double t[N + 1];
        exp_expand<double, N>(t, x0);
        emit_record("exp_expand", N, {x0}, std::vector<double>(t, t + N + 1));
    }

    template <int N>
    static void emit_log_expand(double x0) {
        double t[N + 1];
        log_expand<double, N>(t, x0);
        emit_record("log_expand", N, {x0}, std::vector<double>(t, t + N + 1));
    }

    template <int N>
    static void emit_pow_expand(double x0, double a) {
        double t[N + 1];
        pow_expand<double, N>(t, x0, a);
        emit_record("pow_expand", N, {x0, a}, std::vector<double>(t, t + N + 1));
    }

    template <int N>
    static void emit_sqrt_expand(double x0) {
        double t[N + 1];
        sqrt_expand<double, N>(t, x0);
        emit_record("sqrt_expand", N, {x0}, std::vector<double>(t, t + N + 1));
    }

    template <int N>
    static void emit_cbrt_expand(double x0) {
        double t[N + 1];
        cbrt_expand<double, N>(t, x0);
        emit_record("cbrt_expand", N, {x0}, std::vector<double>(t, t + N + 1));
    }

    template <int N>
    static void emit_gauss_expand(double a) {
        double t[N + 1];
        gauss_expand<double, N>(t, a);
        emit_record("gauss_expand", N, {a}, std::vector<double>(t, t + N + 1));
    }

    template <int N>
    static void emit_erf_expand(double a) {
        double t[N + 1];
        erf_expand<double, N>(t, a);
        emit_record("erf_expand", N, {a}, std::vector<double>(t, t + N + 1));
    }

    // --- ctaylor_mul records ---

    template <int NVAR>
    static void emit_mul_record(const double* a, const double* b) {
        const int SIZE = 1 << NVAR;
        ctaylor<double, NVAR> ta, tb;
        for (int i = 0; i < SIZE; i++) ta.c[i] = a[i];
        for (int i = 0; i < SIZE; i++) tb.c[i] = b[i];
        ctaylor<double, NVAR> tc = ta * tb;

        std::vector<double> inputs(2 * SIZE);
        for (int i = 0; i < SIZE; i++) inputs[i] = a[i];
        for (int i = 0; i < SIZE; i++) inputs[SIZE + i] = b[i];
        std::vector<double> coeffs(tc.c, tc.c + SIZE);
        emit_record("mul", NVAR, inputs, coeffs);
    }

    int main() {
        // ---- *_expand records: 3 inputs × 7 orders × 8 fns = 168 ----

        const double inv_inputs[] = {0.1, 1.0, 10.0};
        const double exp_inputs[] = {-1.0, 0.0, 2.0};
        const double log_inputs[] = {0.1, 1.0, 10.0};

        const double erf_inputs[] = {-1.0, 0.0, 1.0};
        // pow: 3 (x0, a) pairs
        struct PowInput { double x0, a; };
        const PowInput pow_inputs[] = {{1.0, 0.5}, {2.0, 1.5}, {10.0, -1.0}};

        // Unroll per order 0..=6. Template instantiation via manual 7-way switch.
        #define EMIT_EXPAND(FN_NAME, INPUTS_ARR) \
            for (auto& x : INPUTS_ARR) { \
                emit_ ## FN_NAME ## _expand<0>(x); \

                emit_ ## FN_NAME ## _expand<3>(x); \
                emit_ ## FN_NAME ## _expand<4>(x); \
                emit_ ## FN_NAME ## _expand<5>(x); \

        EMIT_EXPAND(inv,  inv_inputs);
        EMIT_EXPAND(exp,  exp_inputs);
        EMIT_EXPAND(log,  log_inputs);
        EMIT_EXPAND(sqrt, sqrt_inputs);
        EMIT_EXPAND(cbrt, cbrt_inputs);
        EMIT_EXPAND(gauss, gauss_inputs);
        EMIT_EXPAND(erf,  erf_inputs);
        for (auto& p : pow_inputs) {
            emit_pow_expand<0>(p.x0, p.a);
            emit_pow_expand<1>(p.x0, p.a);
            emit_pow_expand<2>(p.x0, p.a);
            emit_pow_expand<3>(p.x0, p.a);
            emit_pow_expand<4>(p.x0, p.a);
            emit_pow_expand<5>(p.x0, p.a);
            emit_pow_expand<6>(p.x0, p.a);
        }

        // ---- ctaylor_mul records: 50 seeds × 5 N × 6 ops = 1500 ----
        // We only emit for the "mul" op here; later plans extend to add/sub/...
        // per the inline sampling-strategy note above. Phase 1 Plan 05 ships mul
        // fixtures at n_var ∈ {0, 1, 2, 3, 4} — 50 seeds × 5 N = 250 mul records.
        // (Other ops follow in plan 06 via the same driver extension.)

        std::mt19937_64 rng(0x1234abcdULL);
        std::uniform_real_distribution<double> dist(-10.0, 10.0);

        for (int n_var = 0; n_var <= 4; n_var++) {

            for (int seed_i = 0; seed_i < 50; seed_i++) {

                for (int i = 0; i < SIZE; i++) a[i] = dist(rng);
                for (int i = 0; i < SIZE; i++) b[i] = dist(rng);

                switch (n_var) {
                    case 0: emit_mul_record<0>(a.data(), b.data()); break;
                    case 1: emit_mul_record<1>(a.data(), b.data()); break;
                    case 2: emit_mul_record<2>(a.data(), b.data()); break;

        // Total target: 168 expand + 250 mul (Phase 1 Plan 05) + 1250 other (Phase 1 Plan 06) = 1668.
        // Phase 1 Plan 05 emits 168 + 250 = 418 records; Plan 06 extends to 1668.

        return 0;
    }
    ```

    **Note:** The C++ driver uses `printf("%.17g", ...)` — f64-round-trip format. The Rust side parses into f64 via `str::parse::<f64>` which accepts "%.17g" values verbatim.

    **Alternative approach (cleaner):** have the driver emit a binary format (raw f64 values) instead of text. Concerns:
    - Text is simpler for debugging (can inspect records manually).
    - Binary is immune to locale/FP-printf quirks.

    Recommend **text for Phase 1** per the readability concern; switch to binary in a future iteration if performance matters.

    **1d. `xtask/src/bin/regen_ad_fixtures.rs`** — full populated binary:

    ```rust
    //! Compile the C++ driver, run it, parse its stdout into FixtureRecords,
    //! serialize to bincode, write the manifest.

    use anyhow::{Context, Result, bail};
    use sha2::{Sha256, Digest};
    use std::collections::BTreeMap;
    use std::fs;
    use std::io::Read;
    use std::path::PathBuf;
    use std::process::{Command, Stdio};

    // Re-export the shared schema.
    #[path = "../fixtures.rs"]
    mod fixtures;
    use fixtures::{FixtureRecord, FixturesManifest};

    fn project_root() -> Result<PathBuf> {
        let manifest = std::env::var("CARGO_MANIFEST_DIR")?;
        let xtask_dir = PathBuf::from(manifest);
        let root = xtask_dir.parent().context("xtask has no parent")?.to_path_buf();
        Ok(root)
    }

    fn main() -> Result<()> {
        let root = project_root()?;
        let xcfun_taylor = root.join("xcfun-master/external/upstream/taylor");
        let driver_src = root.join("xtask/assets/regen_ad_fixtures/driver.cpp");
        let out_dir = root.join("target/xtask/fixtures");
        let fixtures_dir = root.join("crates/xcfun-ad/tests/fixtures");
        fs::create_dir_all(&out_dir)?;
        fs::create_dir_all(&fixtures_dir)?;

        // --- Step 1: Compile the driver with cc ---
        let mut build = cc::Build::new();
        build
            .cpp(true)
            .flag_if_supported("-std=c++17")
            .flag_if_supported("-O2")
            // no fast-math, preserve parity
            .flag_if_supported("-fno-fast-math")
            .flag_if_supported("-ffp-contract=off")
            .include(&xcfun_taylor)
            .file(&driver_src)
            .host("x86_64-unknown-linux-gnu")          // canonical CI platform per tolerance table
            .target("x86_64-unknown-linux-gnu")
            .opt_level(2)
            .out_dir(&out_dir);
        let driver_exe = out_dir.join("regen_ad_driver");
        build.try_compile("ad_fixture_driver")
             .context("compile driver.cpp failed — ensure xcfun-master/external/upstream/taylor/ headers are present")?;

        // cc::Build compiles to a static lib, not an exe. Re-invoke with a single link step:
        // Alternative: invoke g++/clang++ directly here. Implementation detail —
        // see the comment block below.
        //
        // For Phase 1, we use the simpler approach: invoke g++ directly to get an
        // executable (cc crate is oriented toward linking into Rust bins, not
        // standalone C++ executables).
        {
            // Prefer CXX env var if set, fall back to c++ / g++ / clang++.
            let compiler = std::env::var("CXX").unwrap_or_else(|_| "c++".into());
            let status = Command::new(&compiler)
                .args([
                    "-std=c++17",
                    "-O2",
                    "-fno-fast-math",
                    "-ffp-contract=off",
                    "-I",
                ])
                .arg(xcfun_taylor.to_str().unwrap())
                .arg(&driver_src)
                .arg("-o")
                .arg(&driver_exe)
                .status()
                .context("spawn c++ compiler")?;
            if !status.success() {
                bail!("compiling driver.cpp failed (exit {:?})", status.code());
            }
        }

        // --- Step 2: Run the driver, parse stdout ---
        let mut cmd = Command::new(&driver_exe)
            .stdout(Stdio::piped())
            .spawn()?;
        let mut stdout = cmd.stdout.take().context("no stdout")?;
        let mut buf = String::new();
        stdout.read_to_string(&mut buf)?;
        let status = cmd.wait()?;
        if !status.success() {
            bail!("driver exited non-zero");
        }

        let mut records: Vec<FixtureRecord> = Vec::new();
        for (i, line) in buf.lines().enumerate() {
            if line.is_empty() { continue; }
            let parts: Vec<&str> = line.split(';').collect();
            if parts.len() != 6 {
                bail!("line {}: malformed (got {} fields, want 6): {}", i, parts.len(), line);
            }
            let op = parts[0].to_string();
            let n_var: u8 = parts[1].parse()?;
            let input_count: usize = parts[2].parse()?;
            let inputs: Vec<f64> = parts[3].split(',')
                .map(|s| s.parse::<f64>().map_err(Into::into))
                .collect::<Result<Vec<_>>>()?;
            anyhow::ensure!(inputs.len() == input_count, "input count mismatch");
            let coeff_count: usize = parts[4].parse()?;
            let coeffs: Vec<f64> = parts[5].split(',')
                .map(|s| s.parse::<f64>().map_err(Into::into))
                .collect::<Result<Vec<_>>>()?;
            anyhow::ensure!(coeffs.len() == coeff_count, "coeff count mismatch");

        let mut expand_records: Vec<FixtureRecord> = Vec::new();

        for rec in &records {
            if rec.op == "mul" {
                mul_records.push(rec.clone());
            } else if rec.op.ends_with("_expand") {
                expand_records.push(rec.clone());
            }
        }

        fs::write(fixtures_dir.join("mul.bincode"), bincode::serialize(&mul_records)?)?;
        fs::write(fixtures_dir.join("expand.bincode"), bincode::serialize(&expand_records)?)?;

        // --- Step 4: Write the JSON manifest with content hash + per-op counts ---
        let mut hasher = Sha256::new();
        for fname in &["ctaylor.hpp", "ctaylor_math.hpp", "tmath.hpp"] {
            let path = xcfun_taylor.join(fname);
            let contents = fs::read(&path).with_context(|| format!("read {:?}", path))?;
            hasher.update(&contents);
        }
        let sha = hasher.finalize();
        let xcfun_version_git_sha = format!("{:x}", sha);

        let mut per_op_counts: BTreeMap<String, usize> = BTreeMap::new();
        for rec in &records {
            *per_op_counts.entry(rec.op.clone()).or_insert(0) += 1;
        }

        let manifest = FixturesManifest {
            xcfun_version_git_sha,
            generated_at: chrono::Utc::now().to_rfc3339(),
            per_op_counts,
            total_records: records.len(),
            driver_commit: Command::new("git")
                .args(["rev-parse", "HEAD"])
                .output()
                .ok()
                .and_then(|o| String::from_utf8(o.stdout).ok())
                .map(|s| s.trim().to_string()),
        };
        fs::write(
            fixtures_dir.join("fixtures.json"),
            serde_json::to_string_pretty(&manifest)?,
        )?;

        println!("regen-ad-fixtures: wrote {} mul records, {} expand records, manifest sha={}",
                 mul_records.len(), expand_records.len(), &manifest.xcfun_version_git_sha[..16]);
        Ok(())
    }
    ```

    ```rust
    //!
    //! Loads `tests/fixtures/mul.bincode` (generated by `cargo run -p xtask --bin
    //! regen-ad-fixtures`) and runs every record through the cubecl `ctaylor_mul`

    use cubecl::prelude::*;
    use cubecl_cpu::CpuRuntime;
    use serde::{Deserialize, Serialize};
    use xcfun_ad::for_tests::cpu_client;
    use xcfun_ad::ctaylor_rec::mul::ctaylor_mul;

    // Keep in sync with xtask/src/fixtures.rs
    #[derive(Serialize, Deserialize, Debug, Clone)]
    struct FixtureRecord {
        op: String,
        n_var: u8,
        inputs: Vec<f64>,
        coeffs: Vec<f64>,
    }

    // Per-n_var kernel wrappers so the const generic is captured at build time.
    #[cube(launch_unchecked)]
    fn mul_kernel(a: &Array<f64>, b: &Array<f64>, out: &mut Array<f64>, #[comptime] n: u32) {
        ctaylor_mul::<f64>(a, b, out, n);
    }

    fn run_mul_record(rec: &FixtureRecord) -> Vec<f64> {
        let client = cpu_client();
        let size = 1_usize << rec.n_var;
        assert_eq!(rec.inputs.len(), 2 * size, "fixture inputs length mismatch");

        let a = &rec.inputs[..size];
        let b = &rec.inputs[size..2 * size];
        let a_h = client.create(f64::as_bytes(a));
        let b_h = client.create(f64::as_bytes(b));
        let out_h = client.empty(size * core::mem::size_of::<f64>());

        unsafe {
            mul_kernel::launch_unchecked::<CpuRuntime>(
                client,
                CubeCount::Static(1, 1, 1),
                CubeDim::new(1, 1, 1),
                ArrayArg::from_raw_parts::<f64>(&a_h, size, 1),
                ArrayArg::from_raw_parts::<f64>(&b_h, size, 1),
                ArrayArg::from_raw_parts::<f64>(&out_h, size, 1),
                rec.n_var as u32,
            );
        }

        let bytes = client.read_one(out_h.binding());
        f64::from_bytes(&bytes).to_vec()
    }

        let bytes: &[u8] = include_bytes!("fixtures/mul.bincode");
    ```
