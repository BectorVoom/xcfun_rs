    //! `inv_expand` — Taylor series of `1 / (a + x)` in `x`, around `x = 0`.
    //!
    //! Port of `xcfun-master/external/upstream/taylor/tmath.hpp:124-129`.
    //!
    //! # C++ source (tmath.hpp:124-129)
    //!
    //! ```cpp
    //! template <class T, int Ndeg> static void inv_expand(T * t, const T & a) {
    //!   assert(a != 0 && "1/(a+x) not analytic at a = 0");
    //!   t[0] = 1/a;
    //!   for (int i = 1; i <= Ndeg; i++)
    //!     t[i] = -t[i - 1] * t[0];
    //! }
    //! ```
    //!
    //! # Identity
    //!
    //! `1 / (a + x) = sum_{i>=0} (-1)^i / a^{i+1} * x^i`.
    //! Equivalently: `t[0] = 1/a`, `t[i] = -t[i-1] * t[0]`.
    //!
    //! # Precondition
    //!
    //! `a != 0`. Checked via `assert!` active in release (D-11).

    use cubecl::prelude::*;

    #[cube]
    pub fn inv_expand<F: Float>(t: &mut Array<F>, a: F, #[comptime] n: u32) {
        // tmath.hpp:125 — precondition guard.
        // NOTE: If cubecl 0.10-pre.3's #[cube] macro rejects host-style assert!,
        // fall back to a host-side debug_assert! before launch + add a comment
        // explaining the fallback.
        assert!(a != F::new(0.0));

        // tmath.hpp:126 — t[0] = 1 / a
        let t0 = F::new(1.0) / a;
        t[0] = t0;

        // tmath.hpp:127-128 — t[i] = -t[i - 1] * t[0]
        #[unroll]
        for i in 1_u32..=n {
            let prev = t[i - 1];
            let neg_prev = F::new(0.0) - prev;
            t[i] = neg_prev * t0;
        }
    }
    ```

    //!   T ifac = 1;
    //!   t[0] = exp(x0);
    //!   for (int i = 1; i <= Ndeg; i++) {

    //!
    //! # Precondition
    //!
    //! None (exp is entire on the reals).

    use cubecl::prelude::*;

    #[cube]
    pub fn exp_expand<F: Float>(t: &mut Array<F>, x0: F, #[comptime] n: u32) {

    **1e. `crates/xcfun-ad/src/expand/log.rs`:**

    READ tmath.hpp:142-151 DIRECTLY before writing this. The sign pattern is:
    ```
    t[i] = x0inv^i / i * sign(i)
    ```
    where `sign(i) = (2 * (i & 1) - 1)`. Test values of `i`:
      - `i=1`: `(2 * 1) - 1 = 1` → `t[1] = +x0inv / 1`
      - `i=2`: `(2 * 0) - 1 = -1` → `t[2] = -x0inv^2 / 2`
      - `i=3`: `(2 * 1) - 1 = 1` → `t[3] = +x0inv^3 / 3`

    That matches the Taylor series of `log(x0 + x) = log(x0) + sum_{i>=1} (-1)^(i+1) / (i * x0^i) * x^i`. Cross-check by expanding `log(1 + y) = y - y^2/2 + y^3/3 - ...` at `x0 = 1`, `y = x/x0`.

    ```rust
    //! `log_expand` — Taylor series of `log(x0 + x)` in `x`, around `x = 0`.
    //!
    //! Port of `xcfun-master/external/upstream/taylor/tmath.hpp:142-151`.
    //!
    //! # C++ source (tmath.hpp:142-151)
    //!
    //! [PASTE THE ACTUAL C++ BLOCK — read tmath.hpp directly]
    //!
    //! # Identity
    //!
    //! `log(x0 + x) = log(x0) + sum_{i>=1} (-1)^(i+1) / (i * x0^i) * x^i`.
    //!
    //! # Precondition
    //!
    //! `x0 > 0`. Checked via `assert!` active in release (D-11).

    use cubecl::prelude::*;

    #[cube]
    pub fn log_expand<F: Float>(t: &mut Array<F>, x0: F, #[comptime] n: u32) {
        assert!(x0 > F::new(0.0));

        t[0] = F::log(x0);
        let x0inv = F::new(1.0) / x0;
        let mut x0inv_pow_i = F::new(1.0);

        // Exact C++ operation order from tmath.hpp:147-150
        #[unroll]
        for i in 1_u32..=n {
            x0inv_pow_i = x0inv_pow_i * x0inv;
            let i_f = F::cast_from(i);
            // sign = 2 * (i & 1) - 1. In F: compute (i & 1) as host-side #[comptime] value.
            // cubecl comptime supports bitwise ops on u32; prefer this form:
            let sign_int = 2_i32 * ((i as i32) & 1) - 1;
            let sign_f = F::cast_from(sign_int);
            // Operation order: (x0inv_pow_i / i) * sign
            let div = x0inv_pow_i / i_f;
            t[i] = div * sign_f;
        }
    }
    ```

    If cubecl 0.10-pre.3 rejects `(i as i32)` inside a comptime-ranging for loop, the sign computation must move to a different primitive — e.g., `F::new(if (i & 1) == 1 { 1.0 } else { -1.0 })` inside a comptime branch. Document any workaround in a file comment.
