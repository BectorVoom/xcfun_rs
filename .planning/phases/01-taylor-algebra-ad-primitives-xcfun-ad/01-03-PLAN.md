// ctaylor.hpp:85-92 — N = 0 base
template <class T> struct ctaylor_rec<T, 0> {
  static void mul(T * dst, const T * x, const T * y) { dst[0] += x[0] * y[0]; }
  static void mul_set(T * dst, const T * x, const T * y) { dst[0] = x[0] * y[0]; }
  static void multo(T * dst, const T * y) { dst[0] *= y[0]; }
  static void multo_skipconst(T * dst, const T * y) { dst[0] = 0; }
  static T eval(const T * ct, const T * x) { return ct[0]; }
  static void compose(T * res, const T * x, const T coeff[]) { res[0] = coeff[0]; }
};

// ctaylor.hpp:94-116 — N = 1
template <class T> struct ctaylor_rec<T, 1> {
  static void mul(T * dst, const T * x, const T * y) {
    dst[0] += x[0] * y[0];
    dst[1] += x[0] * y[1] + x[1] * y[0];
  }
  static void mul_set(T * dst, const T * x, const T * y) {
    dst[0] = x[0] * y[0];
    dst[1] = x[0] * y[1] + x[1] * y[0];
  }
  static void multo(T * dst, const T * y) {
    dst[1] = dst[1] * y[0] + dst[0] * y[1];
    dst[0] *= y[0];
  }
  static void multo_skipconst(T * dst, const T * y) {
    dst[1] = dst[0] * y[1];
    dst[0] = 0;
  }
  static T eval(const T * ct, const T * x) { return ct[0] + x[0] * ct[1]; }
  static void compose(T * res, const T * x, const T coeff[]) {
    res[0] = coeff[0];
    res[1] = coeff[1] * x[1];
  }
};

// ctaylor.hpp:118-152 — N = 2
template <class T> struct ctaylor_rec<T, 2> {
  static void mul(T * dst, const T * x, const T * y) {
    dst[0] += x[0] * y[0];
    dst[1] += x[0] * y[1] + x[1] * y[0];
    dst[2] += x[0] * y[2] + x[2] * y[0];
    dst[3] += x[0] * y[3] + x[3] * y[0] + x[1] * y[2] + x[2] * y[1];
  }
  static void mul_set(T * dst, const T * x, const T * y) {
    dst[0] = x[0] * y[0];
    dst[1] = x[0] * y[1] + x[1] * y[0];
    dst[2] = x[0] * y[2] + x[2] * y[0];
    dst[3] = x[0] * y[3] + x[3] * y[0] + x[1] * y[2] + x[2] * y[1];
  }
  static void multo(T * dst, const T * y) {
    dst[3] = dst[0] * y[3] + dst[3] * y[0] + dst[1] * y[2] + dst[2] * y[1];
    dst[2] = dst[0] * y[2] + dst[2] * y[0];
    dst[1] = dst[0] * y[1] + dst[1] * y[0];
    dst[0] = dst[0] * y[0];
  }
  static void multo_skipconst(T * dst, const T * y) {
    dst[3] = dst[0] * y[3] + dst[1] * y[2] + dst[2] * y[1];
    dst[2] = dst[0] * y[2];
    dst[1] = dst[0] * y[1];
    dst[0] = 0;
  }
  static T eval(const T * ct, const T * x) {
    return ct[0] + x[0] * ct[1] + x[1] * ct[2] + x[1] * x[0] * ct[3];
  }
  static void compose(T * res, const T * x, const T coeff[]) {
    res[0] = coeff[0];
    res[1] = coeff[1] * x[1];
    res[2] = coeff[1] * x[2];
    res[3] = coeff[1] * x[3] + 2 * x[1] * x[2] * coeff[2];
  }
};

    //! Byte-for-byte port of `ctaylor_rec<T, Nvar>` from `xcfun-master/external/upstream/taylor/ctaylor.hpp:41-152`.
    //!
    //! These helpers are the SINGLE load-bearing primitive in Phase 1: a flat triple-loop
    //! rewrite changes the summation order and introduces 1-2 ULP drift per multiply
    //! (Pitfall P3). Do not "simplify".
    //!
    //! Dispatch pattern (B2/W8 revision — LOCKED): two traits keyed off `[f64; 1 << N]`.
    //! CTaylor operations call `<[f64; 1 << N] as MulDispatch>::mul_set(...)` etc.
    //! This is NOT "executor chooses between forms" — the 8 impls below are the contract.

    pub mod mul;
    pub mod multo;
    pub mod compose;

    /// Multiply dispatch trait — one impl per valid `[f64; 1 << N]` array type.
    /// Task 2 provides the 8 impls (N ∈ 0..=7).
    pub trait MulDispatch {
        /// `dst = x * y` (overwrites dst). Delegates to `mul_n{N}_set`.
        fn mul_set(dst: &mut Self, x: &Self, y: &Self);
        /// `dst *= y` (in-place). Delegates to `multo_n{N}`.
        fn multo(dst: &mut Self, y: &Self);
        /// `dst = dst * (y - y[0])` — skip-const variant. Delegates to `multo_skipconst_n{N}`.
        /// Exposed publicly because `compose_n{3..=7}` general recursion calls it.
        fn multo_skipconst(dst: &mut Self, y: &Self);
    }

    /// Compose dispatch trait — one impl per valid `[f64; 1 << N]` array type.
    /// Task 2 provides the 8 impls (N ∈ 0..=7).
    pub trait ComposeDispatch {
        /// `res = sum_i coeff[i] * x^i` assuming `x[0] == 0`. Delegates to `compose_n{N}`.
        /// `coeff` slice length must be ≥ N + 1.
        fn compose(res: &mut Self, x: &Self, coeff: &[f64]);
    }

        let dst_hi_arr: &mut [f64; 4] = dst_hi.try_into().unwrap();

        let y_hi: &[f64; 4] = y[4..].try_into().unwrap();
