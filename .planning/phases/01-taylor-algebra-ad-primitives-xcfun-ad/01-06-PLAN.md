    //! Property tests for `CTaylor<f64, N>` AD engine. D-18, AD-06.
    //!
    //! Nine properties × 10 000 iterations each = 90 000 iterations minimum per CI run.
    //! Shrinking is enabled (default); seed via PROPTEST_SEED env for reproducibility.
    //!
    //! Tolerance ladder:
    //!   - commutativity, identity, inverse, Leibniz on VAR0: bit-exact (to_bits equality)
    //!   - associativity, distributivity: 4 ULP relative
    //!   - exp/log, sqrt², pow inverse roundtrips: 1e-13 relative error

    use proptest::prelude::*;
    use xcfun_ad::{CTaylor, Num, VAR0};

    // Input strategy: uniform f64 in [-100, 100].
    fn f64_bounded() -> impl Strategy<Value = f64> {
        -100.0_f64..100.0_f64
    }

    // N=2 CTaylor: 4 coefficients.
    fn ctaylor_n2() -> impl Strategy<Value = CTaylor<f64, 2>> {
        (f64_bounded(), f64_bounded(), f64_bounded(), f64_bounded())
            .prop_map(|(a, b, c, d)| CTaylor::<f64, 2> { c: [a, b, c, d] })
    }

    fn rel_err(a: f64, b: f64) -> f64 {
        let max = b.abs().max(1.0);
        (a - b).abs() / max
    }

    proptest! {
        #![proptest_config(ProptestConfig { cases: 10_000, .. ProptestConfig::default() })]

        #[test]
        fn add_commutative_n2(a in ctaylor_n2(), b in ctaylor_n2()) {
            let ab = a + b;
            let ba = b + a;
            for i in 0..4 {
                prop_assert_eq!(ab.c[i].to_bits(), ba.c[i].to_bits(),
                    "add non-commutative at coeff {i}: {:?} vs {:?}", ab.c[i], ba.c[i]);
            }
        }

        #[test]
        fn mul_commutative_n2(a in ctaylor_n2(), b in ctaylor_n2()) {
            let ab = a * b;
            let ba = b * a;
            for i in 0..4 {
                prop_assert_eq!(ab.c[i].to_bits(), ba.c[i].to_bits(),
                    "mul non-commutative at coeff {i}");
            }
        }

        #[test]
        fn add_associative_n2(a in ctaylor_n2(), b in ctaylor_n2(), c in ctaylor_n2()) {
            let left = (a + b) + c;
            let right = a + (b + c);
            for i in 0..4 {
                let r = rel_err(left.c[i], right.c[i]);
                let tol = 4.0 * f64::EPSILON;
                prop_assert!(r <= tol, "add non-associative at coeff {i}: rel_err = {r}");
            }
        }

        #[test]
        fn distributive_n2(a in ctaylor_n2(), b in ctaylor_n2(), c in ctaylor_n2()) {
            let lhs = a * (b + c);
            let rhs = a * b + a * c;
            for i in 0..4 {
                let r = rel_err(lhs.c[i], rhs.c[i]);
                let tol = 4.0 * f64::EPSILON;
                prop_assert!(r <= tol, "distribution failed at coeff {i}: rel_err = {r}");
            }
        }

        #[test]
        fn additive_identity_n2(a in ctaylor_n2()) {
            let z = <CTaylor<f64, 2> as Num>::ZERO;
            let r = a + z;
            for i in 0..4 {
                prop_assert_eq!(r.c[i].to_bits(), a.c[i].to_bits(),
                    "additive identity failed at coeff {i}");
            }
        }

        #[test]
        fn multiplicative_identity_n2(a in ctaylor_n2()) {
            let one = <CTaylor<f64, 2> as Num>::ONE;
            let r = a * one;
            for i in 0..4 {
                prop_assert_eq!(r.c[i].to_bits(), a.c[i].to_bits(),
                    "multiplicative identity failed at coeff {i}");
            }
        }

        #[test]
        fn additive_inverse_n2(a in ctaylor_n2()) {
            let r = a - a;
            for i in 0..4 {
                prop_assert_eq!(r.c[i].to_bits(), 0.0_f64.to_bits(),
                    "additive inverse failed at coeff {i}: got {}", r.c[i]);
            }
        }
    }
    ```

    mod roundtrips {
        use super::*;
        use proptest::prelude::*;

        // Constrained CTaylor<f64, 1>: positive c[CNST] for log/sqrt/pow.
        fn positive_ctaylor_n1() -> impl Strategy<Value = CTaylor<f64, 1>> {
            (0.1_f64..100.0, -1.0_f64..1.0)
                .prop_map(|(c0, c1)| CTaylor::<f64, 1> { c: [c0, c1] })
        }

        fn bounded_ctaylor_n1() -> impl Strategy<Value = CTaylor<f64, 1>> {
            (-5.0_f64..5.0, -1.0_f64..1.0)
                .prop_map(|(c0, c1)| CTaylor::<f64, 1> { c: [c0, c1] })
        }

        proptest! {
            #![proptest_config(ProptestConfig { cases: 10_000, .. ProptestConfig::default() })]

            #[test]
            fn exp_log_roundtrip(t in bounded_ctaylor_n1()) {
                let back = t.exp().log();
                for i in 0..2 {
                    let r = super::rel_err(back.c[i], t.c[i]);
                    prop_assert!(r <= 1e-13, "exp/log roundtrip coeff[{}] rel_err = {}, expected {:?}, got {:?}",
                        i, r, t.c[i], back.c[i]);
                }
            }

            #[test]
            fn sqrt_squared_invariance(t in positive_ctaylor_n1()) {
                let s = t.sqrt();
                let back = s * s;
                for i in 0..2 {
                    let r = super::rel_err(back.c[i], t.c[i]);
                    prop_assert!(r <= 1e-13, "sqrt² coeff[{}] rel_err = {}", i, r);
                }
            }

            #[test]
            fn pow_inverse(t in positive_ctaylor_n1(), a in -2.0_f64..2.0) {
                let pos = Num::pow(t, CTaylor::<f64, 1>::from_scalar(a));
                let neg = Num::pow(t, CTaylor::<f64, 1>::from_scalar(-a));
                let product = pos * neg;
                let one = <CTaylor<f64, 1> as Num>::ONE;
                for i in 0..2 {
                    let r = super::rel_err(product.c[i], one.c[i]);
                    prop_assert!(r <= 1e-13, "pow(t,a)*pow(t,-a) coeff[{}] rel_err = {}", i, r);
                }
            }
        }
    }

    mod leibniz {
        use super::*;
        use proptest::prelude::*;
        use xcfun_ad::{CNST, VAR0};

        proptest! {
            #![proptest_config(ProptestConfig { cases: 10_000, .. ProptestConfig::default() })]

            #[test]
            fn leibniz_product_rule_var0(
                a_cnst in -100.0_f64..100.0,
                a_var0 in -100.0_f64..100.0,
                b_cnst in -100.0_f64..100.0,
                b_var0 in -100.0_f64..100.0,
            ) {
                let a = CTaylor::<f64, 1> { c: [a_cnst, a_var0] };
                let b = CTaylor::<f64, 1> { c: [b_cnst, b_var0] };
                let ab = a * b;

                // Leibniz: (a*b)'(x) = a'(x)*b(x) + a(x)*b'(x)
                // In CTaylor: (a*b).c[VAR0] = a.c[CNST]*b.c[VAR0] + a.c[VAR0]*b.c[CNST]
                let left = a.c[CNST] * b.c[VAR0];
                let right = a.c[VAR0] * b.c[CNST];
                let expected = left + right;

                prop_assert_eq!(
                    ab.c[VAR0].to_bits(),
                    expected.to_bits(),
                    "Leibniz VAR0 coeff mismatch: got {:?} ({:#x}), expected {:?} ({:#x})",
                    ab.c[VAR0], ab.c[VAR0].to_bits(),
                    expected, expected.to_bits()
                );
            }
        }
    }
    ```
