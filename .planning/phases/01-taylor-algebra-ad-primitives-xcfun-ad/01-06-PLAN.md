Plan 06 executor can choose either form; document choice in file header.

    ```rust
    /// `ctaylor_powi` — integer-exponent fast path.
    /// Port of `ctaylor_math.hpp:165-178`.
    ///
    /// Branches on `#[comptime] exponent`:
    ///   - exponent > 0: `out = x * x * ... * x` (exponent times)
    ///   - exponent == 0: `out = 1`
    ///   - exponent < 0: `out = ctaylor_pow(x, F::cast_from(exponent))` (falls through to the pow path)
    ///
    /// The C++ version uses a `for (int i = 0; i < n; i++) res *= t;` loop for
    /// positive n. We preserve that left-to-right cumulative multiplication
    /// order via `ctaylor_multo` on the output.
    #[cube]
    pub fn ctaylor_powi<F: Float>(
        x: &Array<F>,
        out: &mut Array<F>,
        scratch: &mut Array<F>,
        #[comptime] exponent: i32,
        #[comptime] n: u32,
    ) {
        // The exponent is comptime, so we can branch at compile time.
        if comptime!(exponent == 0) {
            crate::ctaylor::ctaylor_from_scalar::<F>(F::new(1.0), out, n);
        } else if comptime!(exponent > 0) {
            // out = x
            crate::ctaylor::ctaylor_add::<F>(x, &crate::ctaylor_zero_ref(), out, n);   // out = x + 0
            // Loop: exponent - 1 additional multiplies.
            // cubecl comptime-range-unrolled:
            comptime! {
                for _ in 1..exponent {
                    // out *= x  via ctaylor_multo
                }
            }
            // Since cubecl 0.10-pre.3's comptime for-loop unroll is not fully tested,
            // the executor MAY fall back to an unrolled-explicit specialization per
            // exponent (1, 2, 3, 4, ..., some cap). Limit exponent to ± 16 for Phase 1
            // — matches the C++ primary use case (LDA has pow(rho, 4/3), 4/3, 2/3,
            // etc. — powi is only used for small-integer exponents ≤ 10 in the xcfun
            // functional bodies).
        } else {
            // Negative exponent: C++ delegates to pow(t, double(n)).
            ctaylor_pow::<F>(x, F::cast_from(exponent), out, scratch, n);
        }
    }
    ```

    Add composed-op emitters after the existing expand/mul emitters. The pattern (one per composed fn):

    ```cpp
    template <int NVAR>
    static void emit_composed_record(const char* op_name, const double* x_in, double extra_arg,
                                     bool has_extra) {
        const int SIZE = 1 << NVAR;
        ctaylor<double, NVAR> x;
        for (int i = 0; i < SIZE; i++) x.c[i] = x_in[i];

        ctaylor<double, NVAR> y;
        if (strcmp(op_name, "reciprocal") == 0) y = 1.0 / x;
        else if (strcmp(op_name, "sqrt") == 0)  y = sqrt(x);
        else if (strcmp(op_name, "exp") == 0)   y = exp(x);
        else if (strcmp(op_name, "log") == 0)   y = log(x);
        else if (strcmp(op_name, "pow") == 0)   y = pow(x, extra_arg);
        else if (strcmp(op_name, "erf") == 0)   y = erf(x);
        else if (strcmp(op_name, "asinh") == 0) y = asinh(x);
        else if (strcmp(op_name, "atan") == 0)  y = atan(x);
        else { fprintf(stderr, "unknown op: %s\n", op_name); exit(1); }

        std::vector<double> inputs(x.c, x.c + SIZE);
        if (has_extra) inputs.push_back(extra_arg);
        std::vector<double> coeffs(y.c, y.c + SIZE);

        std::string full_op = std::string("ctaylor_") + op_name;
        emit_record(full_op.c_str(), NVAR, inputs, coeffs);
    }
    ```

    struct ComposedInput { double x_cnst; double x_var0; double extra_arg; bool has_extra; };
    // x.c[0] = x_cnst, x.c[1] = x_var0, x.c[2..=] = 0. Keeps the input set shaped for nvar >= 1.

    // Per-fn inputs — chosen so preconditions (x[CNST] != 0 for reciprocal,
    // x[CNST] > 0 for sqrt/log/pow) are all satisfied across all fns.
    const ComposedInput ci[] = {
        {1.0, 0.5, 0.0, false},   // mild
        {2.0, 1.0, 0.5, true},    // default — extra = 0.5 for pow
        {5.0, -0.1, 1.5, true},   // larger constant, different pow exponent
    };

    const char* composed_ops[] = {
        "reciprocal", "sqrt", "exp", "log", "pow", "erf", "asinh", "atan"
    };

    for (int nvar = 0; nvar <= 3; nvar++) {
        const int SIZE = 1 << nvar;
        for (auto& input : ci) {
            std::vector<double> x_in(SIZE, 0.0);
            x_in[0] = input.x_cnst;
            if (nvar >= 1) x_in[1] = input.x_var0;
            // Higher-order coeffs stay 0 — simplifies reference comparison.

            for (auto* op : composed_ops) {
                // Skip reciprocal if x_cnst == 0 (but we chose all non-zero cnsts above).
                switch (nvar) {
                    case 0: emit_composed_record<0>(op, x_in.data(), input.extra_arg, input.has_extra); break;
                    case 1: emit_composed_record<1>(op, x_in.data(), input.extra_arg, input.has_extra); break;
                    case 2: emit_composed_record<2>(op, x_in.data(), input.extra_arg, input.has_extra); break;
                    case 3: emit_composed_record<3>(op, x_in.data(), input.extra_arg, input.has_extra); break;
                }
            }
        }
    }

    // separately.
    const int powi_exponents[] = {-2, -1, 0, 1, 2, 5, 10};

                    emit_record("ctaylor_powi", 0, inputs, coeffs);
                    break;
                }
                // ... analogous for nvar = 1, 2, 3
            }
        }

        x: &Array<f64>,
        out: &mut Array<f64>,
        scratch: &mut Array<f64>,
        #[comptime] n: u32,

    fn eval_composed(rec: &FixtureRecord) -> Vec<f64> {
        let client = cpu_client();
        let size = 1_usize << rec.n_var;
        let scratch_len = (rec.n_var as usize) + 1;

        let x_in = &rec.inputs[..size];
        let x_h = client.create(f64::as_bytes(x_in));
        let out_h = client.empty(size * core::mem::size_of::<f64>());
        let scratch_h = client.empty(scratch_len * core::mem::size_of::<f64>());

        match rec.op.as_str() {

                    ArrayArg::from_raw_parts::<f64>(&out_h, size, 1),
                    ArrayArg::from_raw_parts::<f64>(&scratch_h, scratch_len, 1),
                    rec.n_var as u32,
                )
            },
            // ... reciprocal, sqrt, log, pow (with extra f64 arg), erf, asinh, atan, powi
            _ => panic!("unknown op in composed.bincode: {}", rec.op),
