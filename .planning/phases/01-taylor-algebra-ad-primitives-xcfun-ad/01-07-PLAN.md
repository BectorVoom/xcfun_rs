    ```rust
    //! Property tests for `xcfun-ad` using the batch-per-property kernel
    //! pattern (CONTEXT.md D-18).
    //!
    //! Each test:
    //! 1. Uses proptest 1.11 to generate 10 000 sample inputs upfront
    //!    (seeded via ProptestConfig + TestRunner).
    //! 2. Packs the inputs into a single cubecl buffer.
    //! 3. Launches ONE cubecl-cpu kernel that evaluates the property across
    //!    all 10 000 points (kernel body parameterized via #[comptime] n_var).
    //! 4. Downloads the result buffer and aggregates pass/fail host-side.
    //!
    //! This preserves the D-18 requirement of ≥ 10 000 iters per property
    //! without paying 10 000 × kernel-launch overhead. PROPTEST_CASES is
    //! honoured as the *reduction knob* for CI runs that need to probe at
    //! lower iteration counts.

    #![cfg(feature = "testing")]

    use cubecl::prelude::*;
    use cubecl_cpu::CpuRuntime;
    use proptest::prelude::*;
    use proptest::strategy::Strategy;
    use proptest::test_runner::TestRunner;
    use xcfun_ad::for_tests::cpu_client;

    const DEFAULT_ITERS: usize = 10_000;

    // Override via PROPTEST_CASES for local experimentation / CI tuning.
    fn iter_count() -> usize {
        std::env::var("PROPTEST_CASES")
            .ok()
            .and_then(|s| s.parse().ok())
            .unwrap_or(DEFAULT_ITERS)
    }

    // Helper: generate `count` samples from a strategy, upfront.
    fn generate<S, T>(strategy: S, count: usize) -> Vec<T>
    where
        S: Strategy<Value = T>,
    {
        let mut runner = TestRunner::deterministic();
        let mut out = Vec::with_capacity(count);
        for _ in 0..count {
            let tree = strategy.new_tree(&mut runner).unwrap();
            out.push(tree.current());
        }
        out
    }
    ```

    Then each property is a `#[test] fn` following the batch-per-property template. Example for commutativity_mul:

    ```rust
    #[cube(launch_unchecked)]
    fn commutativity_mul_kernel(
        a_all: &Array<f64>,       // length K * 4 (K batches, N=1, 2 inputs × 2 coeffs)
        b_all: &Array<f64>,
        diff_all: &mut Array<f64>, // length K * 2 — (a*b - b*a).c [CNST, VAR0]
        #[comptime] k: u32,
    ) {
        #[unroll]
        for i in 0..k {
            // Extract a, b at index i
            let a0 = a_all[i * 2 + 0];
            let a1 = a_all[i * 2 + 1];
            let b0 = b_all[i * 2 + 0];
            let b1 = b_all[i * 2 + 1];

            // Compute (a * b).c via ctaylor_rec::mul::ctaylor_mul_set_n1:
            //   dst[0] = a0 * b0
            //   dst[1] = a0 * b1 + a1 * b0
            let ab0 = a0 * b0;
            let ab1 = a0 * b1 + a1 * b0;

            // (b * a).c:
            let ba0 = b0 * a0;
            let ba1 = b0 * a1 + b1 * a0;

            diff_all[i * 2 + 0] = ab0 - ba0;
            diff_all[i * 2 + 1] = ab1 - ba1;
        }
    }

    #[test]
    fn commutativity_mul_batched_10k() {
        let iters = iter_count();
        let strategy = (
            prop::collection::vec(-100.0_f64..100.0, 2),
            prop::collection::vec(-100.0_f64..100.0, 2),
        );
        let inputs = generate(strategy, iters);

        let mut a_flat = Vec::with_capacity(iters * 2);
        let mut b_flat = Vec::with_capacity(iters * 2);
        for (a, b) in &inputs {
            a_flat.extend_from_slice(a);
            b_flat.extend_from_slice(b);
        }

        let client = cpu_client();
        let a_h = client.create(f64::as_bytes(&a_flat));
        let b_h = client.create(f64::as_bytes(&b_flat));
        let diff_h = client.empty(iters * 2 * core::mem::size_of::<f64>());

        unsafe {
            commutativity_mul_kernel::launch_unchecked::<CpuRuntime>(
                client,
                CubeCount::Static(1, 1, 1),
                CubeDim::new(1, 1, 1),
                ArrayArg::from_raw_parts::<f64>(&a_h, iters * 2, 1),
                ArrayArg::from_raw_parts::<f64>(&b_h, iters * 2, 1),
                ArrayArg::from_raw_parts::<f64>(&diff_h, iters * 2, 1),
                iters as u32,
            );
        }

        let bytes = client.read_one(diff_h.binding());
        let diff = f64::from_bytes(&bytes);
        for (i, d) in diff.iter().enumerate() {
            assert_eq!(*d, 0.0_f64, "commutativity_mul iter={} (coef {}): diff={}", i / 2, i % 2, d);
        }
    }
    ```

    ```rust
            // kernel is written per-N to unroll the mul into explicit
            // coefficient-wise ops indexed by `i * size + j`.
            // [Executor: confirm cubecl Array sub-slice support; else specialize
            //  the kernel body per-N using inline coefficient access.]

    fn bench_mul(c: &mut Criterion) {
        let mut group = c.benchmark_group("ctaylor_mul");
        let client = cpu_client();

                    bencher.iter(|| {
                        let a_h = client.create(f64::as_bytes(&a_in));
                        let b_h = client.create(f64::as_bytes(&b_in));

                            mul_batch_kernel::launch_unchecked::<CpuRuntime>(
                                client,
    ```
