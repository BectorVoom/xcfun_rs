    //! Baseline criterion benchmark for `CTaylor::<f64, N>::mul_assign` at N ∈ {2, 3, 4, 5, 6}.
    //! D-19: Phase 1 baseline only; no regression gate in v1.

    use criterion::{black_box, criterion_group, criterion_main, BenchmarkId, Criterion};
    use xcfun_ad::CTaylor;

    fn seeded_ctaylor<const N: usize>() -> CTaylor<f64, N>
    where xcfun_ad::Bound: xcfun_ad::ValidN<N>,
    {
        let mut c = [0.0_f64; 1 << N];
        // Seeded values from rand-like but deterministic arithmetic.
        for i in 0..(1 << N) {
            c[i] = 0.3 + (i as f64) * 0.17;
        }
        CTaylor { c }
    }

    fn bench_mul_assign<const N: usize>(c: &mut Criterion, label: &str)
    where xcfun_ad::Bound: xcfun_ad::ValidN<N>,
          [f64; 1 << N]: xcfun_ad::ctaylor_rec::MulDispatch,
    {
        let a = seeded_ctaylor::<N>();
        let b = seeded_ctaylor::<N>();
        c.bench_with_input(
            BenchmarkId::new("CTaylor::mul_assign", label),
            &N,
            |bencher, _n| {
                bencher.iter(|| {
                    let mut x = black_box(a);
                    let y = black_box(b);
                    x *= y;
                    black_box(x)
                });
            },
        );
    }

    fn mul_assign_bench_group(c: &mut Criterion) {
        let mut group = c.benchmark_group("mul_assign_per_N");
        // Use discrete benches keyed by N.
        bench_mul_assign::<2>(c, "N=2");
        bench_mul_assign::<3>(c, "N=3");
        bench_mul_assign::<4>(c, "N=4");
        bench_mul_assign::<5>(c, "N=5");
        bench_mul_assign::<6>(c, "N=6");
        group.finish();
    }

    criterion_group!(benches, mul_assign_bench_group);
    criterion_main!(benches);
    ```

    Note on trait bounds: since `MulDispatch` is required, each call site must satisfy it. For N ∈ 2..=6 this should already hold from Plan 03's impls.

    The `--test` flag runs each benchmark once as a smoke test rather than a full measurement loop (fast CI-friendly sanity check). The full measurement runs via `cargo bench -p xcfun-ad --bench mul_bench` without `--test`.

    //! Baseline criterion benchmark for composed `exp`, `log`, `pow` on `CTaylor<f64, 4>`.
    //! D-19.

    use criterion::{black_box, criterion_group, criterion_main, Criterion};
    use xcfun_ad::{CTaylor, Num};

    fn positive_seeded_n4() -> CTaylor<f64, 4> {
        let mut c = [0.0_f64; 16];
        c[0] = 1.5;   // positive CNST so log/sqrt/pow are well-defined
        for i in 1..16 {
            c[i] = 0.01 * ((i as f64) - 8.0);
        }
        CTaylor { c }
    }

    fn bench_exp(c: &mut Criterion) {
        let t = positive_seeded_n4();
        c.bench_function("ctaylor_exp_N4", |b| {
            b.iter(|| black_box(t).exp());
        });
    }

    fn bench_log(c: &mut Criterion) {
        let t = positive_seeded_n4();
        c.bench_function("ctaylor_log_N4", |b| {
            b.iter(|| black_box(t).log());
        });
    }

    fn bench_pow(c: &mut Criterion) {
        let t = positive_seeded_n4();
        let a = 0.75;
        c.bench_function("ctaylor_pow_N4", |b| {
            b.iter(|| crate::ctaylor_pow_wrap(black_box(t), a));
        });
    }

    // Local trampoline — `pow` in the Num trait takes `Self` for the exponent,
    // but the composed `ctaylor_pow(t, f64)` is the direct scalar-exponent path we want to bench.
    fn ctaylor_pow_wrap(t: CTaylor<f64, 4>, a: f64) -> CTaylor<f64, 4> {
        xcfun_ad::math::ctaylor_pow(t, a)
    }

    criterion_group!(benches, bench_exp, bench_log, bench_pow);
    criterion_main!(benches);
    ```
