    ```rust
    //! Property tests for `xcfun-ad` using the batch-per-property kernel
    //! pattern (CONTEXT.md D-18).
    //!
    //! Each test:
    //! 1. Uses proptest 1.11 to generate 10 000 sample inputs upfront
    //!    (seeded via ProptestConfig + TestRunner).
    //! 2. Packs the inputs into a single cubecl buffer.
    //! 3. Launches ONE cubecl-cpu kernel that evaluates the property across
    //!    all 10 000 points (kernel body parameterized via #[comptime] n_var).
    //! 4. Downloads the result buffer and aggregates pass/fail host-side.
    //!
    //! This preserves the D-18 requirement of ≥ 10 000 iters per property
    //! without paying 10 000 × kernel-launch overhead. PROPTEST_CASES is
    //! honoured as the *reduction knob* for CI runs that need to probe at
    //! lower iteration counts.

    #![cfg(feature = "testing")]

    use cubecl::prelude::*;
    use cubecl_cpu::CpuRuntime;
    use proptest::prelude::*;
    use proptest::strategy::Strategy;
    use proptest::test_runner::TestRunner;
    use xcfun_ad::for_tests::cpu_client;

    const DEFAULT_ITERS: usize = 10_000;

    // Override via PROPTEST_CASES for local experimentation / CI tuning.
    fn iter_count() -> usize {
        std::env::var("PROPTEST_CASES")
            .ok()
            .and_then(|s| s.parse().ok())
            .unwrap_or(DEFAULT_ITERS)
    }

    // Helper: generate `count` samples from a strategy, upfront.
    fn generate<S, T>(strategy: S, count: usize) -> Vec<T>
    where
        S: Strategy<Value = T>,
    {
        let mut runner = TestRunner::deterministic();
        let mut out = Vec::with_capacity(count);
        for _ in 0..count {
            let tree = strategy.new_tree(&mut runner).unwrap();
            out.push(tree.current());
        }
        out
    }
    ```

    ```rust
    #[cube(launch_unchecked)]
    fn commutativity_mul_kernel(
        a_all: &Array<f64>,       // length K * 4 (K batches, N=1, 2 inputs × 2 coeffs)
        b_all: &Array<f64>,
        diff_all: &mut Array<f64>, // length K * 2 — (a*b - b*a).c [CNST, VAR0]
        #[comptime] k: u32,
    ) {
        #[unroll]
        for i in 0..k {
            // Extract a, b at index i
            let a0 = a_all[i * 2 + 0];
            let a1 = a_all[i * 2 + 1];
            let b0 = b_all[i * 2 + 0];
            let b1 = b_all[i * 2 + 1];

            // Compute (a * b).c via ctaylor_rec::mul::ctaylor_mul_set_n1:
            //   dst[0] = a0 * b0
            //   dst[1] = a0 * b1 + a1 * b0
            let ab0 = a0 * b0;
            let ab1 = a0 * b1 + a1 * b0;

            // (b * a).c:
            let ba0 = b0 * a0;
            let ba1 = b0 * a1 + b1 * a0;

            diff_all[i * 2 + 0] = ab0 - ba0;
            diff_all[i * 2 + 1] = ab1 - ba1;
        }
    }

    #[test]
    fn commutativity_mul_batched_10k() {
        let iters = iter_count();
        let strategy = (
            prop::collection::vec(-100.0_f64..100.0, 2),
            prop::collection::vec(-100.0_f64..100.0, 2),
        );
        let inputs = generate(strategy, iters);

        let mut a_flat = Vec::with_capacity(iters * 2);
        let mut b_flat = Vec::with_capacity(iters * 2);
        for (a, b) in &inputs {
            a_flat.extend_from_slice(a);
            b_flat.extend_from_slice(b);
        }

        let client = cpu_client();
        let a_h = client.create(f64::as_bytes(&a_flat));
        let b_h = client.create(f64::as_bytes(&b_flat));
        let diff_h = client.empty(iters * 2 * core::mem::size_of::<f64>());

        unsafe {
            commutativity_mul_kernel::launch_unchecked::<CpuRuntime>(
                client,
                CubeCount::Static(1, 1, 1),
                CubeDim::new(1, 1, 1),
                ArrayArg::from_raw_parts::<f64>(&a_h, iters * 2, 1),
                ArrayArg::from_raw_parts::<f64>(&b_h, iters * 2, 1),
                ArrayArg::from_raw_parts::<f64>(&diff_h, iters * 2, 1),
                iters as u32,
            );
        }

        let bytes = client.read_one(diff_h.binding());
        let diff = f64::from_bytes(&bytes);
        for (i, d) in diff.iter().enumerate() {
            assert_eq!(*d, 0.0_f64, "commutativity_mul iter={} (coef {}): diff={}", i / 2, i % 2, d);
        }
    }
    ```

    ```rust
    #[cube(launch_unchecked)]
    fn mul_batch_kernel(a: &Array<f64>, b: &Array<f64>, out: &mut Array<f64>, #[comptime] n: u32, #[comptime] batch: u32) {
        let i = UNIT_POS;
        let size = 1_u32 << n;

            // In cubecl 0.10-pre.3, subslice on Array may require a dedicated
            // Slice<F> type. If Array sub-slicing is not available, the
            // kernel is written per-N to unroll the mul into explicit
            // coefficient-wise ops indexed by `i * size + j`.
            // [Executor: confirm cubecl Array sub-slice support; else specialize
            //  the kernel body per-N using inline coefficient access.]
            // Simplified placeholder:
            for j in 0..size {
                // Not correct — just a stand-in until cubecl sub-slicing is confirmed.
                out[offset + j] = a[offset + j] * b[offset + j];
            }
        }
    }

    fn bench_mul(c: &mut Criterion) {
        let mut group = c.benchmark_group("ctaylor_mul");
        let client = cpu_client();

        for &n in &[2_u32, 3, 4, 5, 6] {
            for &batch in &[1_usize, 64, 1024] {
                let id = BenchmarkId::new(format!("n{}", n), batch);
                let size = 1_usize << n;
                let total = size * batch;

                // Pre-allocate inputs on device
                let a_in: Vec<f64> = (0..total).map(|i| (i as f64) * 0.1).collect();
                let b_in: Vec<f64> = (0..total).map(|i| (i as f64) * 0.2).collect();

                group.bench_with_input(id, &batch, |bencher, &_batch| {
                    bencher.iter(|| {
                        let a_h = client.create(f64::as_bytes(&a_in));
                        let b_h = client.create(f64::as_bytes(&b_in));
                        let out_h = client.empty(total * core::mem::size_of::<f64>());

                        unsafe {
                            mul_batch_kernel::launch_unchecked::<CpuRuntime>(
                                client,
                                CubeCount::Static(1, 1, 1),
                                CubeDim::new(batch as u32, 1, 1),
                                ArrayArg::from_raw_parts::<f64>(&a_h, total, 1),
                                ArrayArg::from_raw_parts::<f64>(&b_h, total, 1),
                                ArrayArg::from_raw_parts::<f64>(&out_h, total, 1),
                                n,
                                batch as u32,
                            );
                        }

                        black_box(bytes);

        }
        group.finish();
    ```

    ```rust
    //! ROADMAP Phase 1 SC #6: assert that cubecl-cpu's MLIR JIT does NOT emit
    //! FMA (`vfmadd{132,213,231}pd/sd`, `fmadd`, `fma213`, `fma231`) in the
    //! `ctaylor_mul*` symbols when `xcfun-ad` is compiled `--release` with
    //! `-Cllvm-args=-fp-contract=off` in `.cargo/config.toml`.
    //!
    //! Operation (D-02, D-03):
    //!   1. `cargo rustc -p xcfun-ad --release --lib -- --emit=asm`
    //!   2. Locate every `.s` file emitted under `target/release/deps/`.
    //!   3. Split each `.s` into function blocks by symbol label.
    //!   4. Demangle each label via `rustc_demangle::demangle`.
    //!   5. If the demangled name contains `ctaylor_mul`, grep the body for
    //!      forbidden FMA mnemonics (`vfmadd*`, `fmadd`, `fma213`, `fma231`).
    //!   6. If any match → print the offending symbol + line + exit 2 (D-03
    //!      escalation: reopen the hand-Rust alternative).
    //!      Else → exit 0.
    //!
    //! Usage:
    //!   cargo run -p xtask --bin check-no-fma
    //!
    //! CI integration: add as a required gate in the `ci.yml` workflow for
    //! any PR touching `crates/xcfun-ad/src/ctaylor_rec/**` or
    //! `.cargo/config.toml`.

    use anyhow::{bail, Context, Result};
    use std::fs;
    use std::path::PathBuf;
    use std::process::Command;

    const FORBIDDEN_MNEMONICS: &[&str] = &[
        // x86-64 fused multiply-add family
        "vfmadd132pd", "vfmadd213pd", "vfmadd231pd",
        "vfmadd132sd", "vfmadd213sd", "vfmadd231sd",
        "vfmsub132pd", "vfmsub213pd", "vfmsub231pd",
        "vfmsub132sd", "vfmsub213sd", "vfmsub231sd",
        "vfnmadd132pd", "vfnmadd213pd", "vfnmadd231pd",
        "vfnmadd132sd", "vfnmadd213sd", "vfnmadd231sd",
        // Generic / aarch64 spellings
        "fmadd",   // aarch64: fmadd d0, d1, d2, d3
        "fmsub",   // aarch64
        "fnmadd",  // aarch64
        "fnmsub",  // aarch64
        // LLVM-intrinsic-style (should never appear after lowering, belt+suspenders)
        "fma213", "fma231",
    ];

    fn project_root() -> Result<PathBuf> {
        let manifest = std::env::var("CARGO_MANIFEST_DIR")
            .context("CARGO_MANIFEST_DIR not set")?;
        let xtask_dir = PathBuf::from(manifest);
        let root = xtask_dir.parent().context("xtask has no parent")?.to_path_buf();
        Ok(root)
    }

    fn main() -> Result<()> {
        let root = project_root()?;

        // Step 1: cargo rustc --emit=asm
        // NOTE: We DO NOT clear target/ first — letting cargo decide what is
        // fresh keeps the local dev loop fast. CI runs with a clean workspace.
        println!("check-no-fma: emitting asm for xcfun-ad --release ...");
        let status = Command::new("cargo")
            .current_dir(&root)
            .args([
                "rustc",
                "-p", "xcfun-ad",
                "--release",
                "--lib",
                "--features", "cpu",
                "--",
                "--emit=asm",
            ])
            .status()
            .context("spawning cargo rustc")?;
        if !status.success() {
            bail!("cargo rustc --emit=asm failed with exit {:?}", status.code());
        }

        // Step 2: find all .s files under target/release/deps/
        let deps_dir = root.join("target/release/deps");
        let mut asm_files = Vec::new();
        for entry in fs::read_dir(&deps_dir)
            .with_context(|| format!("read_dir {}", deps_dir.display()))? {
            let entry = entry?;
            let path = entry.path();
            if path.extension().and_then(|s| s.to_str()) == Some("s")
                && path.file_stem()
                    .and_then(|s| s.to_str())
                    .map(|s| s.starts_with("xcfun_ad"))
                    .unwrap_or(false)
            {
                asm_files.push(path);
            }
        }
        if asm_files.is_empty() {
            bail!(
                "no xcfun_ad-*.s files found under {}. Did cargo rustc --emit=asm actually run?",
                deps_dir.display()
            );
        }
        println!("check-no-fma: scanning {} asm file(s)", asm_files.len());

        // Step 3..5: parse each .s file, find ctaylor_mul symbols, grep bodies
        let mut violations: Vec<String> = Vec::new();
        for path in &asm_files {
            let contents = fs::read_to_string(path)
                .with_context(|| format!("read {}", path.display()))?;
            scan_asm_file(&contents, path.display().to_string().as_str(), &mut violations);
        }

        // Step 6: fail or pass
        if !violations.is_empty() {
            eprintln!("\ncheck-no-fma: FAIL — FMA mnemonics found on ctaylor_mul path:");
            for v in &violations {
                eprintln!("  {}", v);
            }
            eprintln!(
                "\nCONTEXT.md D-03 escalation: cubecl-cpu's MLIR lowering fused arithmetic\n\
                 into FMA instructions despite -Cllvm-args=-fp-contract=off. The 1e-12\n\
                 parity contract cannot be trusted. Planning must escalate via\n\
                 `PLANNING INCONCLUSIVE` per Phase 1 CONTEXT.md D-03."
            );
            std::process::exit(2);
        }

        println!("check-no-fma: PASS — no FMA mnemonics on ctaylor_mul symbols.");
        Ok(())
    }

    /// Scan a single asm file. Splits by symbol labels (lines ending in `:`
    /// that are not `.` directives), demangles each, and if the demangled
    /// name contains `ctaylor_mul`, greps the symbol's body (up to the next
    /// symbol label or `.size` directive) for forbidden mnemonics.
    fn scan_asm_file(contents: &str, path: &str, violations: &mut Vec<String>) {
        let mut current_sym: Option<String> = None;
        let mut sym_is_ctaylor_mul = false;

        for (lineno, line) in contents.lines().enumerate() {
            let trimmed = line.trim_start();

            // Detect a new symbol label: e.g., "_ZN8xcfun_ad12ctaylor_rec3mul...:"
            if let Some(stripped) = trimmed.strip_suffix(':') {
                if !stripped.starts_with('.') && !stripped.is_empty() {
                    let demangled = rustc_demangle::demangle(stripped).to_string();
                    sym_is_ctaylor_mul = demangled.contains("ctaylor_mul");
                    current_sym = Some(demangled);
                    continue;
                }
            }

            if !sym_is_ctaylor_mul { continue; }

            // Inside a ctaylor_mul symbol — grep for forbidden mnemonics.
            // Mnemonic tokens start after any leading whitespace and end at
            // the first whitespace/comma. We just do a substring check which
            // is conservative (false positives from comments are acceptable —
            // they still make the test fail, and fixing a false positive is
            // cheaper than missing a real FMA).
            for mnemonic in FORBIDDEN_MNEMONICS {
                if line.contains(mnemonic) {
                    violations.push(format!(
                        "{}:{} [{}] {}",
                        path,
                        lineno + 1,
                        current_sym.as_deref().unwrap_or("<unknown>"),
                        line.trim(),
                    ));
                }
            }
        }
    }
    ```
