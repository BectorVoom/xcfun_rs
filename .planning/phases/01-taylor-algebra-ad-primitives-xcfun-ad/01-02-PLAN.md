    //! Taylor series of `1/(a+x)`. Port of `xcfun-master/external/upstream/taylor/tmath.hpp:124-129`.
    //!
    //! C++ recurrence (from tmath.hpp:124-129):
    //! ```cpp
    //! template <class T, int N> static void inv_expand(T * t, const T & a) {
    //!   assert(a != 0 && "1/(a+x) not analytic at a = 0");
    //!   t[0] = 1 / a;
    //!   for (int i = 1; i <= N; i++)
    //!     t[i] = -t[i - 1] * t[0];
    //! }
    //! ```

    /// Fill `t[0..=N]` with Taylor coefficients of `1/(a+x)` where N = t.len() - 1.
    ///
    /// Precondition: `a != 0`. Uses `assert!` (D-11) so the check survives in release
    /// (not `debug_assert!` — see P10 in `.planning/research/PITFALLS.md`).
    pub fn inv_expand(t: &mut [f64], a: f64) {
        assert!(a != 0.0, "1/(a+x) not analytic at a = 0");
        t[0] = 1.0 / a;
        for i in 1..t.len() {
            // C++ line 128: t[i] = -t[i - 1] * t[0]
            // Explicit let (D-08) mirrors C++ expression tree for deterministic sequencing.
            let step = -t[i - 1];
            t[i] = step * t[0];
        }
    }
    ```

    //! Taylor series of `exp(x0+x) = exp(x0)*exp(x)`. Port of tmath.hpp:132-139.
    //!
    //! C++ recurrence:
    //! ```cpp
    //! template <class T, int Ndeg> static void exp_expand(T * t, const T & x0) {
    //!   T ifac = 1;
    //!   t[0] = exp(x0);
    //!   for (int i = 1; i <= Ndeg; i++) {
    //!     ifac *= i;
    //!     t[i] = t[0] / ifac;
    //!   }
    //! }
    //! ```

    pub fn exp_expand(t: &mut [f64], x0: f64) {
        let mut ifac: f64 = 1.0;
        // libm boundary: cfg-gated on std/libm per D-14.
        t[0] = exp_f64(x0);
        for i in 1..t.len() {
            // Preserve exact operator order from tmath.hpp:135-138
            let i_f = i as f64;
            ifac = ifac * i_f;  // D-08: explicit let re-bind to match C++ `ifac *= i`
            t[i] = t[0] / ifac;
        }
    }

    #[inline]
    fn exp_f64(x: f64) -> f64 {
        #[cfg(feature = "std")] { x.exp() }
        #[cfg(all(not(feature = "std"), feature = "libm"))] { libm::exp(x) }
    }
    ```

    //! Taylor series of `log(a+x) = log(1+x/a) + log(a)`. Port of tmath.hpp:142-151.
    //!
    //! C++ recurrence:
    //! ```cpp
    //! template <class T, int N> static void log_expand(T * t, const T & x0) {
    //!   assert(x0 > 0 && "log(x) not real analytic at x <= 0");
    //!   t[0] = log(x0);
    //!   T x0inv = 1 / x0;
    //!   T xn = x0inv;
    //!   for (int i = 1; i <= N; i++) {
    //!     t[i] = (xn / double(i)) * (2 * (i & 1) - 1);
    //!     xn *= x0inv;
    //!   }
    //! }
    //! ```

    pub fn log_expand(t: &mut [f64], x0: f64) {
        assert!(x0 > 0.0, "log(x) not real analytic at x <= 0");
        t[0] = log_f64(x0);
        let x0inv = 1.0 / x0;
        let mut xn = x0inv;
        for i in 1..t.len() {
            // Preserve exact order from tmath.hpp:148
            let i_f = i as f64;
            let sign = (2 * ((i & 1) as i32) - 1) as f64;   // +1 odd, -1 even
            let term = xn / i_f;
            t[i] = term * sign;
            xn = xn * x0inv;
        }
    }

    #[inline]
    fn log_f64(x: f64) -> f64 {
        #[cfg(feature = "std")] { x.ln() }
        #[cfg(all(not(feature = "std"), feature = "libm"))] { libm::log(x) }
    }

    //! Taylor series of `(x0+x)^a = x0^a * (1 + x/x0)^a`. Port of tmath.hpp:154-161.
    //!
    //! C++ recurrence:
    //! ```cpp
    //! template <class T, int N> static void pow_expand(T * t, T x0, T a) {
    //!   if (x0 <= 0)
    //!     assert(x0 > 0 && "pow(x,a) not real analytic at x <= 0");
    //!   t[0] = pow(x0, a);
    //!   T x0inv = 1 / x0;
    //!   for (int i = 1; i <= N; i++)
    //!     t[i] = t[i - 1] * x0inv * (a - i + 1) / i;
    //! }
    //! ```

    pub fn pow_expand(t: &mut [f64], x0: f64, a: f64) {
        assert!(x0 > 0.0, "pow(x,a) not real analytic at x <= 0 (got x0 = {x0})");
        t[0] = pow_f64(x0, a);
        let x0inv = 1.0 / x0;
        for i in 1..t.len() {
            let i_f = i as f64;
            // Preserve exact operator order from tmath.hpp:160:
            //   t[i] = t[i - 1] * x0inv * (a - i + 1) / i
            // Left-to-right: (((t[i-1] * x0inv) * (a - i + 1)) / i)
            // Bind each step as `let` to match C++ traversal (D-08).
            let s1 = t[i - 1] * x0inv;
            let s2 = s1 * (a - i_f + 1.0);
            t[i] = s2 / i_f;
        }
    }

    #[inline]
    fn pow_f64(x: f64, a: f64) -> f64 {
        #[cfg(feature = "std")] { x.powf(a) }
        #[cfg(all(not(feature = "std"), feature = "libm"))] { libm::pow(x, a) }
    }

    //! template <class T, int N> static void sqrt_expand(T * t, const T & x0) {
    //!   assert(x0 > 0 && "sqrt(x) not real analytic at x <= 0");
    //!   t[0] = sqrt(x0);
    //!   T x0inv = 1 / x0;

    //!   for (int i = 1; i <= N; i++)

        let x0inv = 1.0 / x0;
        for i in 1..t.len() {

            let den = 3.0 * i_f;
            let quot = num / den;
            let factor = quot - x0inv;
            t[i] = t[i - 1] * factor;
        }
    }

    fn cbrt_f64(x: f64) -> f64 {
        #[cfg(feature = "std")] { x.cbrt() }
        #[cfg(all(not(feature = "std"), feature = "libm"))] { libm::cbrt(x) }
    }
