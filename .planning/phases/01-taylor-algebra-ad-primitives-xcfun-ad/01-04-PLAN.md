    ```rust
    /// Truncated 1D convolution: `z[i] = sum_{j=0..=i} x[j] * y[i-j]`.
    /// Port of tmath.hpp:37-43 (`tfuns::mul`).
    ///
    /// ```cpp
    /// static void mul(T * z, const T * x, const T * y) {
    ///   for (int i = 0; i <= N; i++) {
    ///     z[i] = x[0] * y[i];
    ///     for (int j = 1; j <= i; j++)
    ///       z[i] += x[j] * y[i - j];
    ///   }
    /// }
    /// ```
    #[cube]
    pub fn tfuns_mul<F: Float>(z: &mut Array<F>, x: &Array<F>, y: &Array<F>, #[comptime] n: u32) {
        #[unroll]
        for i in 0_u32..=n {
            let mut acc = x[0] * y[i];
            #[unroll]
            for j in 1_u32..=i {
                acc = acc + x[j] * y[i - j];
            }
            z[i] = acc;
        }
    }
    ```

    **tfuns_multo** (tmath.hpp:45-51): in-place `z *= x`. DESCENDING write order to preserve aliasing.

    ```rust
    /// In-place multiply `z *= x`. Port of tmath.hpp:45-51 (`tfuns::multo`).
    ///
    /// ```cpp
    /// static void multo(T * z, const T * x) {
    ///   for (int i = N; i >= 0; i--) {
    ///     z[i] = x[0] * z[i];
    ///     for (int j = 1; j <= i; j++)
    ///       z[i] += x[j] * z[i - j];
    ///   }
    /// }
    /// ```
    ///
    /// **Write order is descending** — reading `z[i-j]` while writing `z[i]`
    /// requires `z[i-j]` to hold its pre-multiply value. The C++ descending
    /// loop preserves this; a flat ascending loop would corrupt intermediates.
    #[cube]
    pub fn tfuns_multo<F: Float>(z: &mut Array<F>, x: &Array<F>, #[comptime] n: u32) {
        #[unroll]
        for k in 0_u32..=n {
            let i = n - k;   // i counts down from n to 0
            let mut acc = x[0] * z[i];
            #[unroll]
            for j in 1_u32..=i {
                acc = acc + x[j] * z[i - j];
            }
            z[i] = acc;
        }
    }
    ```

    /// Port of tmath.hpp:53-56 (`tfuns::integrate`).
    ///
    /// ```cpp
    /// static void integrate(T * x) {
