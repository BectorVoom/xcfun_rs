    ```rust
    //! cargo xtask regen-ad-fixtures
    //!
    //! Compiles xtask/assets/regen_ad_fixtures/driver.cpp linked against
    //! xcfun-master/external/upstream/taylor/*.hpp, runs it, captures stdout as
    //! JSON records, serialises to bincode + fixtures.json manifest, writes to
    //! crates/xcfun-ad/tests/fixtures/.

    use anyhow::{Context, Result};
    use serde::{Deserialize, Serialize};
    use sha2::{Digest, Sha256};
    use std::{fs, path::PathBuf, process::Command};

    #[derive(Debug, Serialize, Deserialize)]
    struct FixtureRecord {
        op: String,
        n_var: u8,
        inputs: Vec<f64>,
        coeffs: Vec<f64>,
    }

    #[derive(Debug, Serialize, Deserialize)]
    struct FixtureManifest {
        xcfun_version_git_sha: String,
        generator_version: String,
        record_count: usize,
        fixture_files: Vec<String>,
    }

    fn main() -> Result<()> {
        let repo_root = PathBuf::from(env!("CARGO_MANIFEST_DIR")).parent().unwrap().to_path_buf();
        let fixtures_dir = repo_root.join("crates/xcfun-ad/tests/fixtures");
        fs::create_dir_all(&fixtures_dir)?;

        // Step 1: compute sha256 of xcfun-master taylor/*.hpp
        let hash = compute_upstream_hash(&repo_root)?;

        // Step 2: compile driver.cpp via cc
        let driver_src = repo_root.join("xtask/assets/regen_ad_fixtures/driver.cpp");
        let driver_bin = compile_driver(&driver_src, &repo_root)?;

        // Step 3: run driver, parse JSON records from stdout
        let output = Command::new(&driver_bin).output()
            .with_context(|| format!("running {:?}", driver_bin))?;
        anyhow::ensure!(output.status.success(), "driver failed: {}", String::from_utf8_lossy(&output.stderr));
        let records: Vec<FixtureRecord> = serde_json::from_slice(&output.stdout)?;
        anyhow::ensure!(records.len() >= 1668, "Expected ≥ 1668 records, got {}", records.len());

        // Step 4: partition by op family
        let (expand_records, ctaylor_records): (Vec<_>, Vec<_>) = records.iter().cloned().partition(|r| r.op.ends_with("_expand"));
        let (composed_records, ctaylor_records): (Vec<_>, Vec<_>) = ctaylor_records.into_iter().partition(|r| r.op.starts_with("ctaylor_"));

        // Step 5: serialise each family to bincode
        let mut fixture_files = Vec::new();
        write_bincode(&fixtures_dir.join("expand.bincode"), &expand_records)?;
        fixture_files.push("expand.bincode".to_string());
        write_bincode(&fixtures_dir.join("ctaylor_ops.bincode"), &ctaylor_records)?;
        fixture_files.push("ctaylor_ops.bincode".to_string());
        write_bincode(&fixtures_dir.join("composed.bincode"), &composed_records)?;
        fixture_files.push("composed.bincode".to_string());

        // Step 6: write manifest
        let manifest = FixtureManifest {
            xcfun_version_git_sha: hash,
            generator_version: env!("CARGO_PKG_VERSION").to_string(),
            record_count: records.len(),
            fixture_files,
        };
        fs::write(fixtures_dir.join("fixtures.json"), serde_json::to_string_pretty(&manifest)?)?;

        eprintln!("Wrote {} records to {:?}", records.len(), fixtures_dir);
        Ok(())
    }

    fn compute_upstream_hash(repo_root: &std::path::Path) -> Result<String> {
        let mut hasher = Sha256::new();
        let taylor_dir = repo_root.join("xcfun-master/external/upstream/taylor");
        let mut files: Vec<_> = fs::read_dir(&taylor_dir)?
            .filter_map(|e| e.ok())
            .filter(|e| e.path().extension().map_or(false, |x| x == "hpp"))
            .map(|e| e.path())
            .collect();
        files.sort();
        for f in files {
            hasher.update(fs::read(&f)?);
        }
        Ok(format!("{:x}", hasher.finalize()))
    }

    fn compile_driver(src: &std::path::Path, repo_root: &std::path::Path) -> Result<PathBuf> {
        // Use cc::Build to compile driver.cpp. Place the output binary under OUT_DIR.
        // cc normally builds a static library; for a standalone binary we use tools::Tool or cc::Build::compile_intermediates.
        // Simplest path: drive a system C++ compiler directly via std::process::Command.
        let out_dir = repo_root.join("target/xtask-cache");
        fs::create_dir_all(&out_dir)?;
        let out_bin = out_dir.join(if cfg!(windows) { "driver.exe" } else { "driver" });
        let include = repo_root.join("xcfun-master/external/upstream/taylor");
        let status = Command::new(std::env::var("CXX").unwrap_or_else(|_| "c++".into()))
            .args(["-std=c++14", "-O0", "-Wall"])
            .arg("-I").arg(&include)
            .arg(src)
            .arg("-o").arg(&out_bin)
            .status()?;
        anyhow::ensure!(status.success(), "C++ driver compile failed");
        Ok(out_bin)
    }

    fn write_bincode(path: &std::path::Path, records: &[FixtureRecord]) -> Result<()> {
        let bytes = bincode::serialize(records)?;
        fs::write(path, bytes)?;
        Ok(())
    }
    ```

    Driver skeleton (key sections — executor fleshes out the 10 `*_expand` cases including B1's atan/asinh and the 9 `ctaylor` ops):

    ```cpp
    // xtask/assets/regen_ad_fixtures/driver.cpp
    //
    // Emits FixtureRecord JSON records to stdout for the Phase 1 golden parity gate.
    // Linked against xcfun-master/external/upstream/taylor/{ctaylor.hpp, ctaylor_math.hpp, tmath.hpp}.
    //
    // Output format (JSON array on stdout):
    //   [
    //     { "op": "mul", "n_var": 2, "inputs": [...], "coeffs": [...] },
    //     { "op": "exp_expand", "n_var": 3, "inputs": [0.1], "coeffs": [...] },
    //     ...
    //   ]

    #include "ctaylor.hpp"
    #include "ctaylor_math.hpp"
    #include "tmath.hpp"
    #include <cstdio>
    #include <cstdint>
    #include <cstring>
    #include <random>
    #include <string>
    #include <vector>

    // Deterministic seed: 0x1234abcd per research §Sampling strategy.
    static std::mt19937_64 rng(0x1234abcd);
    static std::uniform_real_distribution<double> uni(-10.0, 10.0);
    static std::uniform_real_distribution<double> nonzero(0.5, 2.0);

    static bool first_record = true;

    static void emit_record(const char * op, int n_var,
                            const double * inputs, size_t n_in,
                            const double * coeffs, size_t n_co) {
        if (!first_record) std::fputc(',', stdout);
        first_record = false;
        std::printf("{\"op\":\"%s\",\"n_var\":%d,\"inputs\":[", op, n_var);
        for (size_t i = 0; i < n_in; ++i) {
            if (i) std::fputc(',', stdout);
            std::printf("%.17g", inputs[i]);
        }
        std::fputs("],\"coeffs\":[", stdout);
        for (size_t i = 0; i < n_co; ++i) {
            if (i) std::fputc(',', stdout);
            std::printf("%.17g", coeffs[i]);
        }
        std::fputs("]}", stdout);
    }

    // Emit 3 inputs × 7 orders × 10 functions = 210 expand records (B1 revision — atan/asinh included).
    template<int N>
    static void emit_expand_row(double x0, double a_for_pow) {
        double t[N+1];

        { inv_expand<double, N>(t, x0);    double in[]={x0};            emit_record("inv_expand",   N, in, 1, t, N+1); }
        { exp_expand<double, N>(t, x0);    double in[]={x0};            emit_record("exp_expand",   N, in, 1, t, N+1); }
        { log_expand<double, N>(t, x0);    double in[]={x0};            emit_record("log_expand",   N, in, 1, t, N+1); }
        { pow_expand<double, N>(t, x0, a_for_pow); double in[]={x0, a_for_pow}; emit_record("pow_expand", N, in, 2, t, N+1); }
        { sqrt_expand<double, N>(t, x0);   double in[]={x0};            emit_record("sqrt_expand",  N, in, 1, t, N+1); }
        { cbrt_expand<double, N>(t, x0);   double in[]={x0};            emit_record("cbrt_expand",  N, in, 1, t, N+1); }
        { gauss_expand<double, N>(t, x0);  double in[]={x0};            emit_record("gauss_expand", N, in, 1, t, N+1); }
        { erf_expand<double, N>(t, x0);    double in[]={x0};            emit_record("erf_expand",   N, in, 1, t, N+1); }
        // B1 revision: atan_expand + asinh_expand included — needed by Plan 05 Num::atan/asinh.
        { atan_expand<double, N>(t, x0);   double in[]={x0};            emit_record("atan_expand",  N, in, 1, t, N+1); }
        { asinh_expand<double, N>(t, x0);  double in[]={x0};            emit_record("asinh_expand", N, in, 1, t, N+1); }
    }

    // Helper for ctaylor mul/multo/add/sub/neg/div coverage at N in {0,1,2,3,4}.
    template<int Nvar>
    static void emit_ctaylor_ops(int seed_count) {
        enum { SZ = 1 << Nvar };
        for (int s = 0; s < seed_count; ++s) {
            double a_coeffs[SZ], b_coeffs[SZ];
            for (int i = 0; i < SZ; ++i) { a_coeffs[i] = uni(rng); b_coeffs[i] = uni(rng); }
            // Avoid zero constant term in b for div
            if (std::fabs(b_coeffs[0]) < 0.1) b_coeffs[0] = nonzero(rng);

            ctaylor<double, Nvar> a, b;
            std::memcpy(a.c, a_coeffs, sizeof a_coeffs);
            std::memcpy(b.c, b_coeffs, sizeof b_coeffs);

            double inputs[2 * SZ];
            std::memcpy(inputs, a_coeffs, sizeof a_coeffs);
            std::memcpy(inputs + SZ, b_coeffs, sizeof b_coeffs);

            // mul
            {
                ctaylor<double, Nvar> c;
                ctaylor_rec<double, Nvar>::mul_set(c.c, a.c, b.c);
                emit_record("mul", Nvar, inputs, 2 * SZ, c.c, SZ);
            }
            // add
            {
                ctaylor<double, Nvar> c = a + b;
                emit_record("add", Nvar, inputs, 2 * SZ, c.c, SZ);
            }
            // sub
            {
                ctaylor<double, Nvar> c = a - b;
                emit_record("sub", Nvar, inputs, 2 * SZ, c.c, SZ);
            }
            // neg
            {
                ctaylor<double, Nvar> c = -a;
                emit_record("neg", Nvar, a_coeffs, SZ, c.c, SZ);
            }
            // multo (CTaylor::MulAssign)
            {
                ctaylor<double, Nvar> c = a;
                ctaylor_rec<double, Nvar>::multo(c.c, b.c);
                emit_record("multo", Nvar, inputs, 2 * SZ, c.c, SZ);
            }
            // div
            {
                ctaylor<double, Nvar> c = a / b;
                emit_record("div", Nvar, inputs, 2 * SZ, c.c, SZ);
            }
        }
    }

    // Composed ops (ctaylor_math.hpp) at N = 4, varying c[CNST] and rest seeded.
    template<int Nvar>
    static void emit_composed_ops(int seed_count) {
        enum { SZ = 1 << Nvar };
        for (int s = 0; s < seed_count; ++s) {
            double t_coeffs[SZ];
            t_coeffs[0] = 0.5 + (s * 0.7);       // positive c[CNST] for log/sqrt/pow
            for (int i = 1; i < SZ; ++i) t_coeffs[i] = 0.01 * uni(rng);  // small perturbations

            ctaylor<double, Nvar> t;
            std::memcpy(t.c, t_coeffs, sizeof t_coeffs);

            { auto r = exp(t);         emit_record("ctaylor_exp", Nvar, t_coeffs, SZ, r.c, SZ); }
            { auto r = log(t);         emit_record("ctaylor_log", Nvar, t_coeffs, SZ, r.c, SZ); }
            { auto r = sqrt(t);        emit_record("ctaylor_sqrt", Nvar, t_coeffs, SZ, r.c, SZ); }
            { auto r = pow(t, 0.33);   double inp[SZ+1]; std::memcpy(inp, t_coeffs, sizeof t_coeffs); inp[SZ]=0.33;
                                       emit_record("ctaylor_pow", Nvar, inp, SZ+1, r.c, SZ); }
            { auto r = 1.0 / t;        emit_record("ctaylor_reciprocal", Nvar, t_coeffs, SZ, r.c, SZ); }
            { auto r = erf(t);         emit_record("ctaylor_erf", Nvar, t_coeffs, SZ, r.c, SZ); }
            { auto r = atan(t);        emit_record("ctaylor_atan", Nvar, t_coeffs, SZ, r.c, SZ); }
            { auto r = asinh(t);       emit_record("ctaylor_asinh", Nvar, t_coeffs, SZ, r.c, SZ); }
            { auto r = pow(t, 3);      double inp[SZ+1]; std::memcpy(inp, t_coeffs, sizeof t_coeffs); inp[SZ]=3.0;
                                       emit_record("ctaylor_powi", Nvar, inp, SZ+1, r.c, SZ); }
        }
    }

    int main() {
        std::fputc('[', stdout);

        // 210 expand records (B1 revision — 10 funcs × 3 inputs × 7 orders)
        const double xs[3] = {0.1, 1.0, 10.0};
        const double as[3] = {0.5, -1.0/3.0, 4.0/3.0};
        for (int xi = 0; xi < 3; ++xi) {
            emit_expand_row<0>(xs[xi], as[xi]);
            emit_expand_row<1>(xs[xi], as[xi]);
            emit_expand_row<2>(xs[xi], as[xi]);
            emit_expand_row<3>(xs[xi], as[xi]);
            emit_expand_row<4>(xs[xi], as[xi]);
            emit_expand_row<5>(xs[xi], as[xi]);
            emit_expand_row<6>(xs[xi], as[xi]);
        }

        // 1500 ctaylor op records: 50 seeds × 5 N × 6 ops.
        emit_ctaylor_ops<0>(50);
        emit_ctaylor_ops<1>(50);
        emit_ctaylor_ops<2>(50);
        emit_ctaylor_ops<3>(50);
        emit_ctaylor_ops<4>(50);

        // Composed ops coverage for Wave 2 Plan 05: 20 seeds × N=4 × 9 ops ≈ 180 records.
        emit_composed_ops<4>(20);

        std::fputc(']', stdout);
        std::fputc('\n', stdout);
        return 0;
    }
    ```

    ```rust
        pub coeffs: Vec<f64>,
    }

    pub struct FixtureManifest {

        pub generator_version: String,
        pub record_count: usize,
        pub fixture_files: Vec<String>,
    }

        PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("tests/fixtures")

    pub fn load_manifest() -> FixtureManifest {
        let path = fixtures_dir().join("fixtures.json");
        let bytes = fs::read(&path).expect("fixtures.json missing — run `cargo xtask regen-ad-fixtures`");
        serde_json::from_slice(&bytes).expect("fixtures.json malformed")
    }

    pub fn load_fixtures(filename: &str) -> Vec<FixtureRecord> {
        let path = fixtures_dir().join(filename);

        bincode::deserialize(&bytes).unwrap_or_else(|e| panic!("failed to deserialize {}: {}", filename, e))
    }
    ```
