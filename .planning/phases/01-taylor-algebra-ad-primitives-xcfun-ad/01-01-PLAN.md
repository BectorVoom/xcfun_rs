    ```rust
    impl<const N: usize> CTaylor<f64, N>
    where
        Bound: ValidN<N>,
    {
        /// Constant (no variables). Mirrors `ctaylor(const T & c0)` at ctaylor.hpp:179-187.
        pub fn from_scalar(c0: f64) -> Self {
            let mut c = [0.0_f64; 1 << N];
            c[0] = c0;
            Self { c }
        }

        /// Seeded at variable slot `var` (bit-flag — must be exactly one of VAR0..VAR{N-1}).
        /// Mirrors `ctaylor(const T & c0, int var)` at ctaylor.hpp:188-198.
        /// Precondition: `var` has exactly one bit set and that bit is < N.
        /// Uses `assert!` (D-11) so the check survives in release.
        pub fn from_variable(c0: f64, var: usize) -> Self {
            assert!(var.count_ones() == 1, "from_variable: var must have exactly one bit set, got {var:#b}");
            assert!(var < (1 << N), "from_variable: var bit {var} out of range for N={N}");
            let mut c = [0.0_f64; 1 << N];
            c[0] = c0;
            c[var] = 1.0;
            Self { c }
        }
    }
    ```
